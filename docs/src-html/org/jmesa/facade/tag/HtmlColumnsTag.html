001    package org.jmesa.facade.tag;
002    
003    import static org.jmesa.facade.tag.TagUtils.createInstance;
004    
005    import java.util.Collection;
006    import java.util.List;
007    import java.util.Map;
008    
009    import javax.servlet.jsp.tagext.SimpleTagSupport;
010    
011    import org.jmesa.core.CoreContext;
012    import org.jmesa.util.ItemUtils;
013    import org.jmesa.util.SupportUtils;
014    import org.jmesa.view.component.Column;
015    import org.jmesa.view.html.HtmlComponentFactory;
016    import org.jmesa.view.html.component.HtmlColumn;
017    import org.jmesa.view.html.component.HtmlColumnsGenerator;
018    import org.jmesa.view.html.component.HtmlRow;
019    import org.jmesa.web.WebContext;
020    
021    /**
022     * Used to generate columns on the fly for the tag library.
023     * 
024     * @since 2.2.1
025     * @author Jeff Johnston
026     */
027    public class HtmlColumnsTag extends SimpleTagSupport {
028    
029        private String htmlColumnsGenerator;
030    
031        public String getHtmlColumnsGenerator() {
032            return htmlColumnsGenerator;
033        }
034    
035        public void setHtmlColumnsGenerator(String htmlColumnsGenerator) {
036            this.htmlColumnsGenerator = htmlColumnsGenerator;
037        }
038    
039        private Object getValue(String property, String var) {
040            Object item = getJspContext().getAttribute(var);
041    
042            if (item == null) {
043                return null;
044            }
045    
046            return ItemUtils.getItemValue(item, property);
047        }
048    
049        /**
050         * @return The list of columns generated on the fly.
051         */
052        private List<HtmlColumn> getColumns(HtmlComponentFactory factory, WebContext webContext, CoreContext coreContext) {
053            HtmlColumnsGenerator columnsGenerator = (HtmlColumnsGenerator) createInstance(getHtmlColumnsGenerator());
054            SupportUtils.setWebContext(columnsGenerator, webContext);
055            SupportUtils.setCoreContext(columnsGenerator, coreContext);
056    
057            return columnsGenerator.getColumns(factory);
058        }
059    
060        /**
061         * Process the list of columns that are generated on the fly.
062         */
063        @Override
064        public void doTag() {
065            TableFacadeTag facadeTag = (TableFacadeTag) findAncestorWithClass(this, TableFacadeTag.class);
066            Collection<Map<String, Object>> pageItems = facadeTag.getPageItems();
067            if (pageItems.size() == 1) {
068                HtmlRow row = facadeTag.getTable().getRow();
069                HtmlComponentFactory factory = facadeTag.getComponentFactory();
070                WebContext webContext = facadeTag.getTableFacade().getWebContext();
071                CoreContext coreContext = facadeTag.getTableFacade().getCoreContext();
072                List<HtmlColumn> columns = getColumns(factory, webContext, coreContext);
073                for (HtmlColumn column : columns) {
074                    column.setGeneratedOnTheFly(true);
075                    TagUtils.validateColumn(this, column.getProperty());
076                    row.addColumn(column);
077                }
078            }
079    
080            HtmlRowTag rowTag = (HtmlRowTag) findAncestorWithClass(this, HtmlRowTag.class);
081            Map<String, Object> pageItem = rowTag.getPageItem();
082    
083            HtmlRow row = facadeTag.getTable().getRow();
084            List<Column> columns = row.getColumns();
085            for (Column column : columns) {
086                HtmlColumn htmlColumn = (HtmlColumn) column;
087                if (htmlColumn.isGeneratedOnTheFly()) {
088                    String property = htmlColumn.getProperty();
089                    if (property != null) {
090                        String var = facadeTag.getVar();
091                        pageItem.put(property, getValue(property, var));
092                    }
093                }
094            }
095        }
096    }
