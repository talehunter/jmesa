041    /**
042     * Utility to first check to see if the object being inspected is an instance of the support class.
043     * If it is then will do a check to see if the object to be injected is already set. If it is not
044     * then it will be injected.
045     * 
046     * @since 2.2
047     * @author Jeff Johnston
048     */
049    public class SupportUtils {
050        private SupportUtils() {}
051    
052        /**
053         * Set the WebContext on the object being inspected if it is not already set.
054         * 
055         * @param obj The object being inspected.
056         * @param webContext The object to be injected.
057         */
058        public static void setWebContext(Object obj, WebContext webContext) {
059            if ((obj instanceof WebContextSupport) && ((WebContextSupport) obj).getWebContext() == null) {
060                ((WebContextSupport) obj).setWebContext(webContext);
061            }
062        }
