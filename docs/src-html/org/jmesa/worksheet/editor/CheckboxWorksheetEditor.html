016    package org.jmesa.worksheet.editor;
017    
018    import org.jmesa.limit.Limit;
019    import org.jmesa.view.html.HtmlBuilder;
020    import org.jmesa.worksheet.WorksheetColumn;
021    import static org.jmesa.worksheet.WorksheetUtils.isRowRemoved;
022    
023    /**
024     * Defines a checkbox for the worksheet editor.
025     * 
026     * @since 2.3
027     * @author Jeff Johnston
028     */
029    public class CheckboxWorksheetEditor extends AbstractWorksheetEditor {
030        public static final String CHECKED = "checked";
031        public static final String UNCHECKED = "unchecked";
032        
033        public Object getValue(Object item, String property, int rowcount) {
034            Object value = null;
035            
036            WorksheetColumn worksheetColumn = getWorksheetColumn(item, property);
037            if (worksheetColumn != null) {
038                value = worksheetColumn.getChangedValue();
039            } else {
040                value = getValueForWorksheet(item, property, rowcount);
041            }
042    
043            return getWsColumn(value, item);
044        }
045        
046        /**
047         * Interpret the value of the column to be checked or unchecked. The acceptable 
048         * values for checked is 'y', 'yes', 'true', and '1'. The acceptable values for 
049         * unchecked is 'n', 'no', 'false' and '0'. 
050         * 
051         * @return Either the value 'checked' or 'unchecked'.
052         */
053        @Override
054        public String getValueForWorksheet(Object item, String property, int rowcount) {
055            Object value = super.getValueForWorksheet(item, property, rowcount);
056            
057            if (value == null) {
058                return UNCHECKED;
059            }
060            
061            String valueToConvert = String.valueOf(value);
062            
063            if (valueToConvert.equalsIgnoreCase("y") ||
064                valueToConvert.equalsIgnoreCase("yes") || 
065                valueToConvert.equalsIgnoreCase("true") || 
066                valueToConvert.equals("1")) {
067                return CHECKED;
068            }
069            
070            if (valueToConvert.equalsIgnoreCase("n") ||
071                valueToConvert.equalsIgnoreCase("no") || 
072                valueToConvert.equalsIgnoreCase("false") || 
073                valueToConvert.equals("0")) {
074                return UNCHECKED;
075            }
076    
077            throw new IllegalStateException("Not able to convert the value for the checkbox.");
078        }
079        
080        protected String getWsColumn(Object value, Object item) {
081            HtmlBuilder html = new HtmlBuilder();
082            
083            Limit limit = getCoreContext().getLimit();
084            
085            html.input().type("checkbox");
086            
087            if (value != null && value.equals(CHECKED)) {
088                html.checked();
089            }
090    
091            if (isRowRemoved(getCoreContext().getWorksheet(), getColumn().getRow(), item)) {
092                html.disabled();
093            } else {
094                html.onclick(getUniquePropertyJavaScript(item) + "jQuery.jmesa.submitWsCheckboxColumn(this,'" + 
095                        limit.getId() + "'," + UNIQUE_PROPERTY + ",'" +
096                        getColumn().getProperty() + "')");
097            }
098            
099            html.end();
100            
101            return html.toString();
102        }
103    }
