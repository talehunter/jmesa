016    package org.jmesa.worksheet.editor;
017    
018    import org.jmesa.view.component.Row;
019    import org.jmesa.view.editor.AbstractCellEditor;
020    import org.jmesa.view.editor.CellEditor;
021    import org.jmesa.worksheet.UniqueProperty;
022    import org.jmesa.worksheet.Worksheet;
023    import org.jmesa.worksheet.WorksheetColumn;
024    import org.jmesa.worksheet.WorksheetRow;
025    
026    /**
027     * The abstraction for the worksheet editors. Custom implementations should extend this.
028     * 
029     * @since 2.3
030     * @author Jeff Johnston
031     */
032    public abstract class AbstractWorksheetEditor extends AbstractCellEditor implements WorksheetEditor {
033        protected String UNIQUE_PROPERTY = "up";
034    
035        private CellEditor cellEditor;
036        
037        /**
038         * @return The wrapped CellEditor.
039         */
040        public CellEditor getCellEditor() {
041            return cellEditor;
042        }
043    
044        /**
045         * @param cellEditor The CellEditor to wrap.
046         */
047        public void setCellEditor(CellEditor cellEditor) {
048            this.cellEditor = cellEditor;
049        }
050        
051        public Object getValueForWorksheet(Object item, String property, int rowcount) {
052            return getCellEditor().getValue(item, property, rowcount);
053        }
054    
055        /**
056         * @param item The Bean or Map.
057         * @param property The column property.
058         * @return The WorksheetColumn for this column.
059         */
060        protected WorksheetColumn getWorksheetColumn(Object item, String property) {
061            Worksheet worksheet = getCoreContext().getWorksheet();
062            if (worksheet == null) {
063                return null;
064            }
065            
066            UniqueProperty uniqueProperty = getColumn().getRow().getUniqueProperty(item);
067            WorksheetRow worksheetRow = worksheet.getRow(uniqueProperty);
068    
069            if (worksheetRow == null) {
070                return null;
071            }
072            
073            return worksheetRow.getColumn(property);
074        }
075        
076        /**
077         * @param item The Bean or Map.
078         * @return The JavaScript for the unique properties.
079         */
080        protected String getUniquePropertyJavaScript(Object item) {
081            Row row = getColumn().getRow();
082            UniqueProperty uniqueProperty = row.getUniqueProperty(item);
083            if (uniqueProperty == null) {
084                return "";
085            }
086    
087            StringBuilder sb = new StringBuilder();
088            sb.append("var " + UNIQUE_PROPERTY + " = {};");
089            sb.append(UNIQUE_PROPERTY + "['" + uniqueProperty.getName() + "']='" + uniqueProperty.getValue() + "';");
090            return sb.toString();
091        }
092    }
