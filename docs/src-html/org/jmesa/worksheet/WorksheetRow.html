016    package org.jmesa.worksheet;
017    
018    import java.io.Serializable;
019    import java.util.Collection;
020    import java.util.LinkedHashMap;
021    import java.util.Map;
022    
023    import org.apache.commons.lang.builder.ToStringBuilder;
024    import org.jmesa.core.message.Messages;
025    
026    /**
027     * <p>
028     * A WorksheetRow contains WorksheetColumns. A row also has a given status. It can be a newly added
029     * row, a modified row, or a deleted row.
030     * </p>
031     *
032     * <p>
033     * If a row is modified and then later deleted (before being saved) then it will be deleted from the
034     * Worksheet. There is no reason for anyone to be aware that a row was almost created. That should
035     * make sense. On the other hand if you delete a row that existed previously (ie, the row exists in
036     * the database) then you have to flag the row as being deleted so that you would know to delete the
037     * row in the database.
038     * </p>
039     *
040     * @since 2.3
041     * @author Jeff Johnston
042     */
043    public class WorksheetRow implements Serializable {
044        private Worksheet worksheet;
045        private UniqueProperty uniqueProperty;
046        private WorksheetRowStatus rowStatus;
047        private Object item;
048    
049        private Map<String, WorksheetColumn> columns = new LinkedHashMap<String, WorksheetColumn>();
050    
051        public WorksheetRow(UniqueProperty uniqueProperty) {
052            this.uniqueProperty = uniqueProperty;
053        }
054    
055        public void setWorksheet(Worksheet worksheet) {
056            this.worksheet = worksheet;
057        }
058    
059        /**
060         * @return Map in which the map keys are the item properties and the map values are the item
061         *         values.
062         */
063        public UniqueProperty getUniqueProperty() {
064            return uniqueProperty;
065        }
066    
067        /**
068         * Add a column to the worksheet row.
069         *
070         * @param column The worksheet row column.
071         */
072        public void addColumn(WorksheetColumn column) {
073            columns.put(column.getProperty(), column);
074            column.setRow(this);
075        }
076    
077        /**
078         * @param property The column property.
079         * @return The worksheet row column.
080         */
081        public WorksheetColumn getColumn(String property) {
082            return columns.get(property);
083        }
084    
085        /**
086         * @return All the row columns in the worksheet.
087         */
088        public Collection<WorksheetColumn> getColumns() {
089            return columns.values();
090        }
091    
092        /**
093         * Remove the specified worksheet row column.
094         *
095         * @param column The worksheet row column to remove.
096         */
097        public void removeColumn(WorksheetColumn column) {
098            columns.remove(column.getProperty());
099        }
100    
101        /**
102         * @return Get the row status.
103         */
104        public WorksheetRowStatus getRowStatus() {
105            return rowStatus;
106        }
107    
108        /**
109         * Set the row status.
110         *
111         * @param rowStatus The row status.
112         */
113        public void setRowStatus(WorksheetRowStatus rowStatus) {
114            this.rowStatus = rowStatus;
115        }
116    
117        /**
118         * @return Is true if any of the row columns contain errors.
119         */
120        public boolean hasErrors() {
121            Collection<WorksheetColumn> values = columns.values();
122            for (WorksheetColumn worksheetColumn : values) {
123                boolean hasError = worksheetColumn.hasError();
124                if (hasError) {
125                    return true;
126                }
127            }
128    
129            return false;
130        }
131    
132        /**
133         * Set the backing worksheet item on the row.
134         *
135         * @param item The backing worksheet item.
136         */
137            public void setItem(Object item) {
138                    this.item = item;
139            }
140    
141        /**
142         * @return Get the backing worksheet item.
143         */
144            public Object getItem() {
145                    return item;
146            }
147    
148        public Messages getMessages() {
149            return worksheet.getMessages();
150        }
151    
152        @Override
153        public String toString() {
154            ToStringBuilder builder = new ToStringBuilder(this);
155            builder.append("uniqueProperty", uniqueProperty);
156            builder.append("columns", columns);
157            return builder.toString();
158        }
159    }
