016    package org.jmesa.view.html.renderer;
017    
018    
019    import org.jmesa.view.editor.CellEditor;
020    import org.jmesa.view.html.HtmlBuilder;
021    import static org.jmesa.view.html.HtmlConstants.CELL_RENDERER_INCLUDE_ID;
022    import org.jmesa.view.html.component.HtmlColumn;
023    import org.jmesa.view.renderer.AbstractCellRenderer;
024    import org.jmesa.worksheet.editor.WorksheetEditor;
025    
026    /**
027     * @since 2.0
028     * @author Jeff Johnston
029     */
030    public class HtmlCellRenderer extends AbstractCellRenderer {
031    
032        public HtmlCellRenderer() {}
033    
034        public HtmlCellRenderer(HtmlColumn column) {
035            setColumn(column);
036        }
037    
038        /**
039         * @deprecated The CellEditor should be set directly on the HtmlColumn.
040         */
041        @Deprecated
042        public HtmlCellRenderer(HtmlColumn column, CellEditor editor) {
043            setColumn(column);
044            setCellEditor(editor);
045        }
046    
047        @Override
048        public HtmlColumn getColumn() {
049            return (HtmlColumn) super.getColumn();
050        }
051    
052        /**
053         * @deprecated Should get/set the value on the HtmlColumn.
054         */
055        @Deprecated
056        public String getStyle() {
057            return getColumn().getStyle();
058        }
059    
060        /**
061         * @deprecated Should get/set the value on the HtmlColumn.
062         */
063        @Deprecated
064        public void setStyle(String style) {
065            getColumn().setStyle(style);
066        }
067    
068        /**
069         * @deprecated Should get/set the value on the HtmlColumn.
070         */
071        @Deprecated
072        public String getStyleClass() {
073            return getColumn().getStyleClass();
074        }
075    
076        /**
077         * @deprecated Should get/set the value on the HtmlColumn.
078         */
079        @Deprecated
080        public void setStyleClass(String styleClass) {
081            getColumn().setStyleClass(styleClass);
082        }
083    
084        /**
085         * @deprecated Should get/set the value on the HtmlColumn.
086         */
087        @Deprecated
088        @Override
089        public CellEditor getCellEditor() {
090            return getColumn().getCellEditor();
091        }
092    
093        @Deprecated
094        @Override
095        public void setCellEditor(CellEditor cellEditor) {
096            getColumn().setCellEditor(cellEditor);
097        }
098    
099        /**
100         * @deprecated Should get/set the value on the HtmlColumn.
101         */
102        @Deprecated
103        public WorksheetEditor getWorksheetEditor() {
104            return getColumn().getWorksheetEditor();
105        }
106    
107        /**
108         * @deprecated Should get/set the value on the HtmlColumn.
109         */
110        @Deprecated
111        public void setWorksheetEditor(WorksheetEditor worksheetEditor) {
112            getColumn().setWorksheetEditor(worksheetEditor);
113        }
114    
115        protected String getId(int rowcount) {
116            if (getCoreContext().getPreference(CELL_RENDERER_INCLUDE_ID).equals("false")) {
117                return null;
118            }
119    
120            String id = getCoreContext().getLimit().getId() + "_column_" + getColumn().getProperty() + "_" + rowcount;
121            id = id.replaceAll("\\.", "_");
122            return id;
123        }
124    
125        public Object render(Object item, int rowcount) {
126            HtmlBuilder html = new HtmlBuilder();
127            html.td(2).id(getId(rowcount));
128            html.width(getColumn().getWidth());
129            html.style(getStyle());
130            html.styleClass(getStyleClass());
131            html.close();
132    
133            String property = getColumn().getProperty();
134            Object value = getCellEditor().getValue(item, property, rowcount);
135            if (value != null) {
136                html.append(value.toString());
137            } else {
138                html.nbsp();
139            }
140    
141            html.tdEnd();
142    
143            return html.toString();
144        }
145    }
