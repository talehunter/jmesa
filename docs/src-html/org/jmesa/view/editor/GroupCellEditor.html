016    package org.jmesa.view.editor;
017    
018    import java.util.List;
019    import org.jmesa.view.component.Column;
020    import org.jmesa.view.component.Row;
021    
022    /**
023     * @since 2.3.2
024     * @author Jeff Johnston
025     */
026    public class GroupCellEditor extends AbstractCellEditor {
027        private static final String LAST_GROUPED_COLUMN = "LAST_GROUPED_COLUMN";
028    
029        private Row currentRow;
030        private Object lastColumnValue;
031        private CellEditor decoratedCellEditor;
032    
033        public GroupCellEditor(CellEditor decoratedCellEditor) {
034            this.decoratedCellEditor = decoratedCellEditor;
035        }
036    
037        public Object getValue(Object item, String property, int rowcount) {
038            this.currentRow = getColumn().getRow();
039    
040            Object columnValue = decoratedCellEditor.getValue(item, property, rowcount);
041    
042            boolean repeating = isRepeating(columnValue);
043            this.lastColumnValue = columnValue;
044    
045            Column column = getColumn();
046            boolean firstColumn = isFirstColumn(column);
047            boolean previousColumnGrouped = isPreviousColumnGrouped(column);
048    
049            // reset at the first column
050            if (firstColumn && !repeating) {
051                setLastGroupedColumn(null);
052            }
053    
054            // check for duplicates
055            if ((firstColumn || previousColumnGrouped) && repeating) {
056                setLastGroupedColumn(column);
057                columnValue = null;
058            }
059    
060            return columnValue;
061        }
062    
063        private boolean isRepeating(Object columnValue) {
064            return String.valueOf(columnValue).equals(String.valueOf(lastColumnValue));
065        }
066    
067        private boolean isPreviousColumnGrouped(Column column) {
068            Column lastGroupedColumn = getLastGroupedColumn();
069            if (lastGroupedColumn == null) {
070                return false;
071            }
072    
073            List<Column> columns = currentRow.getColumns();
074            int position = columns.indexOf(column);
075            int lastGroupedColumnPosition = columns.indexOf(lastGroupedColumn);
076    
077            return lastGroupedColumnPosition == position - 1;
078        }
079    
080        private boolean isFirstColumn(Column column) {
081            Column firstColumn = null;
082    
083            List<Column> columns = currentRow.getColumns();
084            for (Column col : columns) {
085                if (col.getCellRenderer().getCellEditor() instanceof GroupCellEditor) {
086                    firstColumn = col;
087                    break;
088                }
089            }
090    
091            return column.equals(firstColumn);
092        }
093    
094        private Column getLastGroupedColumn() {
095            return (Column) getCoreContext().getAttribute(LAST_GROUPED_COLUMN);
096        }
097    
098        private void setLastGroupedColumn(Column column) {
099            getCoreContext().setAttribute(LAST_GROUPED_COLUMN, column);
100        }
101    }
