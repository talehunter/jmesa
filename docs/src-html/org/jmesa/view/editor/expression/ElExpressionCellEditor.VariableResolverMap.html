016    package org.jmesa.view.editor.expression;
017    
018    import static org.jmesa.util.AssertUtils.notNull;
019    import org.jmesa.view.editor.*;
020    import java.io.StringReader;
021    import java.util.HashMap;
022    import java.util.Map;
023    
024    import javax.servlet.jsp.el.ELException;
025    import javax.servlet.jsp.el.FunctionMapper;
026    import javax.servlet.jsp.el.VariableResolver;
027    
028    import org.apache.commons.el.ExpressionString;
029    import org.apache.commons.el.parser.ELParser;
030    import org.apache.commons.el.parser.ParseException;
031    import org.slf4j.Logger;
032    import org.slf4j.LoggerFactory;
033    
034    /**
035     * Uses commons-el to evaluate EL expressions.
036     * 
037     * @version 2.4
038     * @author bgould
039     */
040    public class ElExpressionCellEditor extends AbstractCellEditor {
041        private Logger logger = LoggerFactory.getLogger(ElExpressionCellEditor.class);
042        private org.apache.commons.el.Logger pLogger = new org.apache.commons.el.Logger(System.err);
043    
044        private String var;
045        private Object template;
046    
047        public ElExpressionCellEditor(Expression expression) {
048            this(expression.getVar(), expression.getTemplate());
049        }
050    
051        public ElExpressionCellEditor(String var, Object template) {
052            notNull("The var is required.", var);
053            this.var = var;
054    
055            notNull("The template is required.", template);
056            this.template = template;
057    
058            try {
059                this.template = new ELParser(new StringReader(String.valueOf(template))).ExpressionString();
060            } catch (ParseException e) {
061                this.template = null;
062                throw new RuntimeException(e);
063            }
064        }
065    
066        public Object getValue(Object item, String property, int rowcount) {
067            Object result = null;
068    
069            try {
070                // ExpressionString is a mixture of template text and EL
071                // expressions; ex. ${lastName}, ${firstName}
072                if (template instanceof ExpressionString) {
073                    result = ((ExpressionString) template).evaluate(getVariableResolver(item), getFunctionMapper(), pLogger);
074    
075                // Expression is a single EL expression with no template text;
076                // ex. ${lastName + ', ' + firstName}
077                } else if (template instanceof org.apache.commons.el.Expression) {
078                    result = ((org.apache.commons.el.Expression) template).evaluate(getVariableResolver(item), getFunctionMapper(), pLogger);
079    
080                // If the expression parsed to a String, it is just template text 
081                } else if (template instanceof String) {
082                    result = template;
083                }
084            } catch (ELException e) {
085                logger.warn("Could not process el expression editor with property " + property, e);
086            }
087    
088            return result;
089        }
090    
091        /**
092         * Creates a VariableResolver based on the current row bean and variable name.
093         * 
094         * @param item The row's backing bean.
095         */
096        protected VariableResolver getVariableResolver(Object item) {
097            Map<String, Object> context = new HashMap<String, Object>();
098            context.put(var, item);
099            return new VariableResolverMap(context);
100        }
101    
102        /**
103         * Override this method to make EL functions available to your expressions.
104         */
105        protected FunctionMapper getFunctionMapper() {
106            return null;
107        }
108    
109        /**
110         * VariableResolver that resolves implicit objects based on a Map.
111         */
112        public static class VariableResolverMap implements VariableResolver {
113    
114            private final Map<?, ?> context;
115    
116            public VariableResolverMap(Map<?, ?> context) {
117                this.context = context;
118            }
119    
120            public Object resolveVariable(String var) {
121                return context.get(var);
122            }
123        }
124    }
