016    package org.jmesa.view.editor.expression;
017    
018    import static org.jmesa.util.AssertUtils.notNull;
019    import org.jmesa.view.editor.*;
020    import org.apache.bsf.BSFException;
021    import org.apache.bsf.BSFManager;
022    import org.jmesa.core.CoreContext;
023    import org.slf4j.Logger;
024    import org.slf4j.LoggerFactory;
025    
026    /**
027     *  Uses the BSF engine to evaluate expressions.
028     *
029     * @version 2.4
030     * @author bgould
031     */
032    public class BsfExpressionCellEditor extends AbstractCellEditor {
033        private Logger logger = LoggerFactory.getLogger(BsfExpressionCellEditor.class);
034    
035        private final Language language;
036        private final String var;
037        private final Object template;
038    
039        public BsfExpressionCellEditor(Expression expression) {
040            this(expression.getLanguage(), expression.getVar(), expression.getTemplate());
041        }
042    
043        public BsfExpressionCellEditor(Language language, String var, Object template) {
044            notNull("The language is required.", language);
045            this.language = language;
046    
047            if (!BSFManager.isLanguageRegistered(language.toString().toLowerCase())) {
048                throw new IllegalArgumentException("The language " + language + " is not supported.");
049            }
050    
051            notNull("The var is required.", var);
052            this.var = var;
053    
054            notNull("The template is required.", template);
055            this.template = template;
056        }
057    
058        public Object getValue(Object item, String property, int rowcount) {
059            Object result = null;
060    
061            BSFManager manager = getBsfManager();
062            try {
063                manager.declareBean(var, item, Object.class);
064                result = manager.eval(language.toString().toLowerCase(), "jmesa", 0, 0, template);
065                manager.undeclareBean(var);
066            } catch (BSFException e) {
067                logger.warn("Could not process bsf expression editor with property " + property, e);
068            } finally {
069                manager.terminate();
070            }
071    
072            return result;
073        }
074    
075        /**
076         * Cache the manager on a per-table basis to insure thread safety.
077         */
078        protected BSFManager getBsfManager() {
079            CoreContext ctx = getCoreContext();
080            BSFManager manager = (BSFManager) ctx.getAttribute("org.jmesa.BSFManager");
081            if (manager == null) {
082                manager = new BSFManager();
083                ctx.setAttribute("org.jmesa.BSFManager", manager);
084            }
085            return manager;
086        }
087    }
