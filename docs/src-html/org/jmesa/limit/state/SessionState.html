016    package org.jmesa.limit.state;
017    
018    import org.jmesa.limit.Limit;
019    import org.slf4j.Logger;
020    import org.slf4j.LoggerFactory;
021    
022    /**
023     * <p>
024     * Pass the stateAttr to find the Limit with the persisted state. The allows a
025     * user to display a table in the way they left it.
026     * </p>
027     * 
028     * @since 2.0
029     * @author Jeff Johnston
030     */
031    public class SessionState extends AbstractState {
032        private Logger logger = LoggerFactory.getLogger(SessionState.class);
033        
034        public Limit retrieveLimit() {
035            String stateAttrValue = getWebContext().getParameter(getStateAttr());
036            if ("true".equalsIgnoreCase(stateAttrValue)) {
037                if (logger.isDebugEnabled()) {
038                    logger.debug("The Limit is being retrieved from the users session.");
039                }
040                return (Limit) getWebContext().getSessionAttribute(getId());
041            }
042            
043            stateAttrValue = (String)getWebContext().getRequestAttribute(getStateAttr());
044            if ("true".equalsIgnoreCase(stateAttrValue)) {
045                if (logger.isDebugEnabled()) {
046                    logger.debug("The Limit is being retrieved from the users session.");
047                }
048                return (Limit) getWebContext().getSessionAttribute(getId());
049            }
050    
051            return null;
052        }
053    
054        public void persistLimit(Limit limit) {
055            if (logger.isDebugEnabled()) {
056                logger.debug("The Limit is being persisted on the users session.");
057            }
058            getWebContext().setSessionAttribute(getId(), limit);
059        }
060    
061        @Override
062        public void setId(String id) {
063            super.setId(id + "_LIMIT");
064        }
065    }
