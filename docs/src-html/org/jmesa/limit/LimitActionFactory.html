016    package org.jmesa.limit;
017    
018    import java.util.Map;
019    
020    import org.apache.commons.lang.StringUtils;
021    import org.apache.commons.lang.builder.ToStringBuilder;
022    import org.slf4j.Logger;
023    import org.slf4j.LoggerFactory;
024    
025    /**
026     * Figure out how the user interacted with the table.
027     *
028     * @since 2.0
029     * @author Jeff Johnston
030     */
031    public class LimitActionFactory {
032        private Logger logger = LoggerFactory.getLogger(LimitActionFactory.class);
033        
034        private final Map<?, ?> parameters;
035        private final String id;
036        private final String prefixId;
037    
038        public LimitActionFactory(String id, Map<?, ?> parameters) {
039            this.id = id;
040            this.parameters = parameters;
041            this.prefixId = id + "_";
042        }
043    
044        public String getId() {
045            return id;
046        }
047    
048        /**
049         * @return The max rows based on what the user selected. A null returned implies the default
050         *         must be used.
051         */
052        public Integer getMaxRows() {
053            String maxRows = LimitUtils.getValue(parameters.get(prefixId + Action.MAX_ROWS.toParam()));
054            if (StringUtils.isNotBlank(maxRows)) {
055                if (logger.isDebugEnabled()) {
056                    logger.debug("Max Rows:" + maxRows);
057                }
058                return Integer.parseInt(maxRows);
059            }
060    
061            return null;
062        }
063    
064        /**
065         * @return The current page based on what the user selected. The default is to return the first
066         *         page.
067         */
068        public int getPage() {
069            String page = LimitUtils.getValue(parameters.get(prefixId + Action.PAGE.toParam()));
070            if (StringUtils.isNotBlank(page)) {
071                if (logger.isDebugEnabled()) {
072                    logger.debug("On Page :" + page);
073                }
074                return Integer.parseInt(page);
075            }
076    
077            if (logger.isDebugEnabled()) {
078                logger.debug("Defaulting to Page 1");
079            }
080    
081            return 1;
082        }
083    
084        public FilterSet getFilterSet() {
085            FilterSet filterSet = new FilterSet();
086    
087            String clear = LimitUtils.getValue(parameters.get(prefixId + Action.CLEAR.toParam()));
088            if (StringUtils.isNotEmpty(clear)) {
089                if (logger.isDebugEnabled()) {
090                    logger.debug("Cleared out the filters.");
091                }
092                return filterSet;
093            }
094    
095            for (Object param : parameters.keySet()) {
096                String parameter = (String) param;
097                if (parameter.startsWith(prefixId + Action.FILTER.toParam())) {
098                    String value = LimitUtils.getValue(parameters.get(parameter));
099                    if (StringUtils.isNotBlank(value)) {
100                        String property = StringUtils.substringAfter(parameter, prefixId + Action.FILTER.toParam());
101                        Filter filter = new Filter(property, value);
102                        filterSet.addFilter(filter);
103                    }
104                }
105            }
106    
107            return filterSet;
108        }
109    
110        public SortSet getSortSet() {
111            SortSet sortSet = new SortSet();
112    
113            for (Object param : parameters.keySet()) {
114                String parameter = (String) param;
115                if (parameter.startsWith(prefixId + Action.SORT.toParam())) {
116                    String value = LimitUtils.getValue(parameters.get(parameter));
117                    if (StringUtils.isNotBlank(value)) {
118                        String position = StringUtils.substringBetween(parameter, prefixId + Action.SORT.toParam(), "_");
119                        String property = StringUtils.substringAfter(parameter, prefixId + Action.SORT.toParam() + position + "_");
120                        Order order = Order.valueOfParam(value);
121                        Sort sort = new Sort(new Integer(position), property, order);
122                        sortSet.addSort(sort);
123                    }
124                }
125            }
126    
127            return sortSet;
128        }
129    
130        /**
131         * @return The current export type based on what the user selected.
132         */
133        public ExportType getExportType() {
134            String exportType = LimitUtils.getValue(parameters.get(prefixId + Action.EXPORT.toParam()));
135            if (StringUtils.isNotBlank(exportType)) {
136                if (logger.isDebugEnabled()) {
137                    logger.debug("ExportType: " + exportType);
138                }
139                ExportType et = ExportType.valueOfParam(exportType);
140                if (et != null) {
141                    return et;
142                }
143    
144                throw new IllegalStateException("Not able to handle the export of type: " + exportType);
145            }
146    
147            return null;
148        }
149    
150        @Override
151        public String toString() {
152            ToStringBuilder builder = new ToStringBuilder(this);
153            builder.append("id", id);
154            builder.append("prefixId", prefixId);
155            if (parameters != null) {
156                parameters.toString();
157            }
158            return builder.toString();
159        }
160    }
