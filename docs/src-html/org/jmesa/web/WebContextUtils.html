016    package org.jmesa.web;
017    
018    import java.util.List;
019    
020    /**
021     * @since 2.0
022     * @author Jeff Johnston
023     */
024    public class WebContextUtils {
025        /**
026         * The value needs to be a String[]. A String, Null, or List will be
027         * converted to a String[]. In addition it will attempt to do a String
028         * conversion for other object types.
029         * 
030         * @param value
031         *            The value to convert to an String[]
032         * @return A String[] value.
033         */
034        public static String[] getValueAsArray(Object value) {
035            if (value == null) {
036                return new String[] {}; // put in a placeholder
037            }
038    
039            if (value instanceof String[]) {
040                return (String[]) value;
041            } else if (value instanceof List) {
042                List<?> valueList = (List<?>) value;
043                return valueList.toArray(new String[valueList.size()]);
044            }
045    
046            return new String[] { value.toString() };
047        }
048    }
