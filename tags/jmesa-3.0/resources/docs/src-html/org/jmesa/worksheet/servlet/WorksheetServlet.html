016    package org.jmesa.worksheet.servlet;
017    
018    import static org.jmesa.core.message.MessagesFactory.getMessages;
019    
020    import java.io.IOException;
021    
022    import javax.servlet.http.HttpServlet;
023    import javax.servlet.http.HttpServletRequest;
024    import javax.servlet.http.HttpServletResponse;
025    
026    import org.jmesa.core.message.Messages;
027    import org.jmesa.web.HttpServletRequestWebContext;
028    import org.jmesa.web.WebContext;
029    import org.jmesa.worksheet.WorksheetUpdater;
030    
031    /**
032     * The default invoker of the worksheet updater.
033     *
034     * @since 2.3
035     * @author Jeff Johnston
036     */
037    public class WorksheetServlet extends HttpServlet {
038        @Override
039        public void doGet(HttpServletRequest request, HttpServletResponse response) {
040            WebContext webContext = new HttpServletRequestWebContext(request);
041            Messages messages = getMessages(webContext);
042            WorksheetUpdater worksheetUpdater = new WorksheetUpdater();
043            String columnStatus = worksheetUpdater.update(messages, webContext);
044            
045            try {
046                response.getWriter().print(columnStatus);
047            } catch (IOException e) {
048                e.printStackTrace();
049            }
050        }
051    
052        @Override
053        public void doPost(HttpServletRequest request, HttpServletResponse response) {
054            doGet(request, response);
055        }
056    }
