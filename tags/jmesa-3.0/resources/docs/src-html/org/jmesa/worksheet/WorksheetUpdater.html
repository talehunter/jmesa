016    package org.jmesa.worksheet;
017    
018    import java.io.UnsupportedEncodingException;
019    import static java.net.URLDecoder.decode;
020    import java.util.Map;
021    
022    import org.apache.commons.lang.StringUtils;
023    import org.jmesa.core.message.Messages;
024    import org.jmesa.web.WebContext;
025    import org.jmesa.worksheet.state.SessionWorksheetState;
026    import org.jmesa.worksheet.state.WorksheetState;
027    
028    /**
029     * Will store the Worksheet object in the users session by the table id. However, once the servlet
030     * is set up the developer will not ever have to deal with the fact that the Worksheet object is in
031     * session. The ajax calls will abstract that out from the html table side. Then the TableFacade
032     * will abstract out the retrieve of the Worksheet in the controller.
033     *
034     * @since 2.3
035     * @author Jeff Johnston
036     */
037    public class WorksheetUpdater {
038        protected static String UNIQUE_PROPERTIES = "up_";
039        protected static String COLUMN_PROPERTY = "cp_";
040        protected static String ORIGINAL_VALUE = "ov_";
041        protected static String CHANGED_VALUE = "cv_";
042    
043        protected static String ERROR_MESSAGE = "em_";
044    
045        protected static String COL_REMOVED = "_rm_";
046        protected static String COL_UPDATED = "_uu_";
047        protected static String COL_HAS_ERROR = "_ue_";
048        protected WorksheetState wst;
049    
050        public String update(Messages messages, WebContext webContext) {
051            Worksheet worksheet = getWorksheet(messages, webContext);
052            WorksheetRow row = getWorksheetRow(worksheet, webContext);
053    
054            WorksheetColumn column = getWorksheetColumn(row, webContext);
055            String columnStatus = validateWorksheet(worksheet, row, column, webContext.getParameter(ERROR_MESSAGE));
056    
057            return columnStatus;
058        }
059    
060        protected Worksheet getWorksheet(Messages messages, WebContext webContext) {
061            wst = getWorksheetState(webContext);
062            Worksheet worksheet = wst.retrieveWorksheet();
063            if (worksheet == null) {
064                    String id = webContext.getParameter("id");
065                worksheet = new Worksheet(id);
066                worksheet.setWebContext(webContext);
067                worksheet.setMessages(messages);
068                wst.persistWorksheet(worksheet);
069            }
070    
071            return worksheet;
072        }
073    
074        protected WorksheetState getWorksheetState(WebContext webContext) {
075            if (wst == null) {
076                    String id = webContext.getParameter("id");
077                    return new SessionWorksheetState(id, webContext);
078            }
079    
080            return wst;
081        }
082    
083        protected WorksheetRow getWorksheetRow(Worksheet worksheet, WebContext webContext) {
084            Map<?, ?> parameters = webContext.getParameterMap();
085            for (Object param : parameters.keySet()) {
086                String parameter = (String) param;
087                if (parameter.startsWith(UNIQUE_PROPERTIES)) {
088                    String value = webContext.getParameter(parameter);
089                    String property = StringUtils.substringAfter(parameter, UNIQUE_PROPERTIES);
090    
091                    UniqueProperty uniqueProperty = new UniqueProperty(property, value);
092                    WorksheetRow row = worksheet.getRow(uniqueProperty);
093                    if (row == null) {
094                        row = new WorksheetRow(uniqueProperty);
095                        row.setRowStatus(WorksheetRowStatus.MODIFY);
096                        worksheet.addRow(row);
097                    }
098    
099                    return row;
100                }
101            }
102    
103            return null;
104        }
105    
106        protected WorksheetColumn getWorksheetColumn(WorksheetRow row, WebContext webContext) {
107            String property = webContext.getParameter(COLUMN_PROPERTY);
108            WorksheetColumn column = row.getColumn(property);
109            if (column == null) {
110                String orginalValue = webContext.getParameter(ORIGINAL_VALUE);
111                try {
112                    orginalValue = decode(orginalValue, "UTF-8");
113                } catch (UnsupportedEncodingException e) {
114                    e.printStackTrace();
115                }
116                column = new WorksheetColumn(property, orginalValue);
117                row.addColumn(column);
118            }
119    
120            String changedValue = webContext.getParameter(CHANGED_VALUE);
121            try {
122                changedValue = decode(changedValue, "UTF-8");
123            } catch (UnsupportedEncodingException e) {
124                e.printStackTrace();
125            }
126            column.setChangedValue(changedValue);
127    
128            return column;
129        }
130    
131        /**
132         * Validate that the columns original value is not the same as the changed value. If it is
133         * then remove the column from the row.
134         */
135        protected String validateWorksheet(Worksheet worksheet, WorksheetRow row, WorksheetColumn column, String errorMessage) {
136            String columnStatus = COL_UPDATED;
137    
138            if (StringUtils.isNotEmpty(errorMessage)) {
139                try {
140                    column.setError(decode(errorMessage, "UTF-8"));
141                } catch (UnsupportedEncodingException e) {
142                    e.printStackTrace();
143                }
144                columnStatus = COL_HAS_ERROR;
145            } else if (column.hasError()) {
146                column.removeError();
147            }
148    
149            if (column.getChangedValue().equals(column.getOriginalValue())) {
150                if (!row.getRowStatus().equals(WorksheetRowStatus.ADD)) {
151                    row.removeColumn(column);
152    
153                    if (row.getColumns().size() == 0) {
154                        worksheet.removeRow(row);
155                    }
156                }
157    
158                if (!columnStatus.equals(COL_HAS_ERROR)) {
159                    columnStatus = COL_REMOVED;
160                }
161            }
162    
163            return columnStatus;
164        }
165    }
