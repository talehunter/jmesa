016    package org.jmesa.limit;
017    
018    import org.apache.commons.lang.StringUtils;
019    import org.jmesa.limit.state.SessionState;
020    import org.jmesa.limit.state.State;
021    import org.jmesa.util.SupportUtils;
022    import org.jmesa.web.WebContext;
023    
024    /**
025     * <p>
026     * Constructs a Limit and a RowSelect object.
027     * </p>
028     *
029     * <p>
030     * An example to create a Limit using the LimitFactory is as follows:
031     * </p>
032     *
033     * <p>
034     * First you need to pass in the table id and the WebContext. The table id is the unique identifier
035     * for the current table being built. The WebContext is an adapter for the servlet request.
036     * </p>
037     *
038     * <pre>
039     * String id = &quot;pres&quot;;
040     * WebContext webContext = new HttpServletRequestWebContext(request);
041     * LimitFactory limitFactory = new LimitFactory(id, webContext);
042     * Limit limit = limitFactory.createLimit();
043     * </pre>
044     *
045     * <p>
046     * Once you have the Limit you still need to create and set a RowSelect object on the Limit. The
047     * RowSelect needs to be added to the Limit so that the row information is available. The catch is
048     * the RowSelect cannot be created until the total rows is known. If you are not manually filtering
049     * and sorting the table then it is easy because the totalRows is just the size of your items
050     * (Collection of Beans or Collection of Maps). However, if you are manually filtering and sorting
051     * the table to bring back one page of data the you first need to use the FilterSet on the Limit to
052     * figure out the total rows.
053     * </p>
054     *
055     * <p>
056     * Either way, once you have the total rows you can now create a RowSelect with the factory. Lastly,
057     * set the RowSelect on the Limit and your done!
058     * </p>
059     *
060     * <pre>
061     * RowSelect rowSelect = limitFactory.createRowSelect(maxRows, totalRows);
062     * limit.setRowSelect(rowSelect);
063     * </pre>
064     *
065     * <p>
066     * Note: if you are not manually filtering and sorting there is a convenience method to create the
067     * Limit and RowSelect at once.
068     * </p>
069     *
070     * <pre>
071     * LimitFactory limitFactory = new LimitFactory(id, webContext);
072     * Limit limit = limitFactory.createLimitAndRowSelect(maxRows, totalRows);
073     * </pre>
074     *
075     * <p>
076     * Also, if you are manually filtering and sorting and are doing an export then you should create
077     * the RowSelect object yourself and set it on the Limit.
078     * </p>
079     *
080     * <pre>
081     * RowSelect rowSelect = new RowSelect(1, totalRows, totalRows);
082     * limit.setRowSelect(rowSelect);
083     * </pre>
084     *
085     * @since 2.0
086     * @author Jeff Johnston
087     */
088    public class LimitFactory {
089        private final LimitActionFactory limitActionFactory;
090        private final String id;
091        private final WebContext webContext;
092        private State state;
093    
094        /**
095         * @param id The unique identifier for the current table being built.
096         * @param webContext The adapter for the servlet request.
097         */
098        public LimitFactory(String id, WebContext webContext) {
099            this.id = id;
100            this.webContext = webContext;
101            this.limitActionFactory = new LimitActionFactory(id, webContext.getParameterMap());
102        }
103    
104        public void setState(State state) {
105            this.state = state;
106        }
107    
108        /**
109         * Utilize the State interface to persist the Limit in the users HttpSession. Will persist the
110         * Limit by the id.
111         *
112         * @param stateAttr The parameter that will be searched to see if the state should be used.
113         * 
114         * @deprecated The State should be set directly on the Limit. This really should not be a part of the interface because
115         *             it is an implementation detail.
116         */
117        @Deprecated
118        public void setStateAttr(String stateAttr) {
119            if (StringUtils.isNotEmpty(stateAttr)) {
120                this.state = new SessionState();
121                SupportUtils.setId(state, id);
122                SupportUtils.setStateAttr(state, stateAttr);
123                SupportUtils.setWebContext(state, webContext);
124            }
125        }
126    
127        /**
128         * <p>
129         * Create a Limit object that is populated with the FilterSet, SortSet, and Export. Be aware
130         * though that the Limit object is still incomplete. You still need to set a RowSelect on the
131         * Limit to make the object complete.
132         * </p>
133         *
134         * <p>
135         * Also note that if you are using the State feature the returned object will be the Limit as
136         * it exists in the State.
137         * </p>
138         *
139         * <p>
140         * One reason to create the Limit separately from the RowSelect is if you are going to manually
141         * filter and sort the table to only return one page of data. If you are doing that then you
142         * should use the FilterSet to manually filter the table to figure out the totalRows.
143         * </p>
144         */
145        public Limit createLimit() {
146            Limit limit = getStateLimit();
147    
148            if (limit != null) {
149                return limit;
150            }
151    
152            limit = new Limit(limitActionFactory.getId());
153    
154            FilterSet filterSet = limitActionFactory.getFilterSet();
155            limit.setFilterSet(filterSet);
156    
157            SortSet sortSet = limitActionFactory.getSortSet();
158            limit.setSortSet(sortSet);
159    
160            ExportType exportType = limitActionFactory.getExportType();
161            limit.setExportType(exportType);
162    
163            if (state != null && !limit.hasExport()) {
164                state.persistLimit(limit);
165            }
166    
167            return limit;
168        }
169    
170        /**
171         * <p>
172         * Create the RowSelect object. This is created with dynamic values for the page and maxRows.
173         * What this means is if the user is paginating or selected a different maxRows on the table
174         * then the RowSelect will be created using those values.
175         * </p>
176         *
177         * <p>
178         * If you do not want the RowSelect to be created based on how the user is interacting with the
179         * table, then do not use this method. Instead you should instantiate the RowSelect object
180         * yourself and set it on the Limit.
181         * </p>
182         *
183         * @param maxRows The maximum page rows that should be displayed on the current page.
184         * @param totalRows The total rows across all the pages.
185         */
186        public RowSelect createRowSelect(int maxRows, int totalRows) {
187            int page = limitActionFactory.getPage();
188    
189            maxRows = getMaxRows(maxRows);
190    
191            return new RowSelect(page, maxRows, totalRows);
192        }
193    
194        /**
195         * @deprecated Because the Limit is injected this is not a very good factory method.
196         */
197        @Deprecated
198        public RowSelect createRowSelect(int maxRows, int totalRows, Limit limit) {
199            RowSelect rowSelect = createRowSelect(maxRows, totalRows);
200            limit.setRowSelect(rowSelect);
201            return rowSelect;
202        }
203    
204        /**
205         * <p>
206         * A convenience method to create the Limit and RowSelect at the same time. This takes in
207         * account whether or not the table is being exported. If the table is being exported a new
208         * RowSelect object is created, starting at page one, and the maxRows and totalRows set to the
209         * value of the totalRows. If the table is not being exported then the RowSelect will be created
210         * based on how the user is interacting with the table. What this means is if the user is
211         * paginating or selected a different maxRows on the table then the RowSelect will be created
212         * using those values.
213         * </p>
214         *
215         * <p>
216         * If you do not want the RowSelect to be created based on how the user is interacting with the
217         * table, then do not use this method. Instead create the Limit using the createLimit() method
218         * and then create a RowSelect object separately.
219         * </p>
220         *
221         * <p>
222         * This method is also not useful if you are using the Limit to filter and sort manually and
223         * only return one page of data. This is because when you are manually filtering you do not know
224         * the totalRows until you use the FilterSet to do the filtering.
225         * </p>
226         *
227         * @param maxRows The maximum page rows that should be displayed on the current page.
228         * @param totalRows The total rows across all the pages.
229         *
230         * @return The created Limit that is populated with the RowSelect object.
231         */
232        public Limit createLimitAndRowSelect(int maxRows, int totalRows) {
233            Limit limit = createLimit();
234    
235            if (limit.hasRowSelect()) {
236                return limit;
237            }
238    
239            if (limit.hasExport()) {
240                RowSelect rowSelect = new RowSelect(1, totalRows, totalRows);
241                limit.setRowSelect(rowSelect);
242            } else {
243                RowSelect rowSelect = createRowSelect(maxRows, totalRows);
244                limit.setRowSelect(rowSelect);
245            }
246    
247            return limit;
248        }
249    
250        private int getMaxRows(int maxRows) {
251            Integer currentMaxRows = limitActionFactory.getMaxRows();
252            if (currentMaxRows == null) {
253                return maxRows;
254            }
255    
256            return currentMaxRows;
257        }
258    
259        private Limit getStateLimit() {
260            if (state != null) {
261                Limit l = state.retrieveLimit();
262                if (l != null) {
263                    return l;
264                }
265            }
266    
267            return null;
268        }
269    }
