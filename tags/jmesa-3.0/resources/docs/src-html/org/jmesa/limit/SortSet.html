016    package org.jmesa.limit;
017    
018    import java.io.Serializable;
019    
020    import java.util.ArrayList;
021    import java.util.Collection;
022    import java.util.Collections;
023    import java.util.Iterator;
024    import java.util.List;
025    
026    import org.apache.commons.lang.builder.ToStringBuilder;
027    
028    /**
029     * <p>
030     * The SortSet is an Collection of Sort objects. A Sort contains an item property and the sort
031     * Order. Or, in other words, it is simply the column that the user is trying to sort in the correct
032     * order.
033     * </p>
034     *
035     * @since 2.0
036     * @author Jeff Johnston
037     */
038    public class SortSet implements Serializable {
039        private List<Sort> sorts;
040    
041        public SortSet() {
042            sorts = new ArrayList<Sort>();
043        }
044    
045        public boolean isSortable() {
046            return isSorted();
047        }
048    
049        /**
050         * @return Is true if there are any columns that need to be sorted.
051         */
052        public boolean isSorted() {
053            return sorts != null && !sorts.isEmpty();
054        }
055    
056        /**
057         * @return The set of Sort objects.
058         */
059        public Collection<Sort> getSorts() {
060            return sorts;
061        }
062    
063        /**
064         * <p>
065         * For a given item property, retrieve the Sort object based on the property.
066         * </p>
067         *
068         * @param property The Sort property, which is also a column property.
069         * @return The Sort object.
070         */
071        public Sort getSort(String property) {
072            for (Iterator<Sort> iter = sorts.iterator(); iter.hasNext();) {
073                Sort sort = iter.next();
074                if (sort.getProperty().equals(property)) {
075                    return sort;
076                }
077            }
078    
079            return null;
080        }
081    
082        /**
083         * For a given property, retrieve the Sort Order.
084         *
085         * @param property The Sort property, which is also a column property.
086         * @return The Sort Order.
087         */
088        public Order getSortOrder(String property) {
089            return getSort(property).getOrder();
090        }
091    
092        /**
093         * <p>
094         * The Sort to add to the set in the given position.
095         * </p>
096         *
097         * @param position The position in the set to add the Sort.
098         * @param property The Sort property, which is also a column property.
099         * @param order The Order to sort the column.
100         */
101        public void addSort(int position, String property, Order order) {
102            addSort(new Sort(position, property, order));
103        }
104    
105        /**
106         * <p>
107         * The Sort to add to the set. Will also set the sort position to be when the sort was added.
108         * </p>
109         *
110         * <p>
111         * For example if there is already a Sort in the set then calling this method will add a Sort to
112         * the end. If you need to include a Sort in a given position of the Set then call the other
113         * convenience method that includes the position.
114         * </p>
115         *
116         * @param property The Sort property, which is also a column property.
117         * @param order The Order to sort the column.
118         */
119        public void addSort(String property, Order order) {
120            addSort(new Sort(sorts.size(), property, order));
121        }
122    
123        /**
124         * @param sort The Sort to add to the set.
125         */
126        public void addSort(Sort sort) {
127            if (sorts.contains(sort)) {
128                sorts.remove(sort);
129            }
130    
131            sorts.add(sort);
132            Collections.sort(sorts);
133        }
134    
135        @Override
136        public String toString() {
137            ToStringBuilder builder = new ToStringBuilder(this);
138    
139            if (sorts != null) {
140                for (Iterator<Sort> iter = sorts.iterator(); iter.hasNext();) {
141                    Sort sort = iter.next();
142                    builder.append(sort.toString());
143                }
144            }
145    
146            return builder.toString();
147        }
148    }
