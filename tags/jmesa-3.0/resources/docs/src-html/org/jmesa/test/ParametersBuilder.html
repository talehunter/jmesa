016    package org.jmesa.test;
017    
018    import static org.jmesa.worksheet.Worksheet.FILTER_WORKSHEET;
019    import static org.jmesa.worksheet.Worksheet.SAVE_WORKSHEET;
020    
021    import java.util.ArrayList;
022    import java.util.List;
023    
024    import static org.jmesa.facade.TableFacadeUtils.TABLE_REFRESHING;
025    import org.jmesa.limit.Action;
026    import org.jmesa.limit.ExportType;
027    import org.jmesa.limit.Order;
028    
029    /**
030     * Build up the parameters that is used by the LimitFactory.
031     * 
032     * @since 2.0
033     * @author Jeff Johnston
034     */
035    public class ParametersBuilder {
036        private final String prefixId;
037        private final Parameters parameters;
038        private int sortCount;
039    
040        public ParametersBuilder(String id, Parameters parameters) {
041            this.prefixId = id + "_";
042            this.parameters = parameters;
043        }
044    
045        public void setPage(int page) {
046            String key = prefixId + Action.PAGE.toParam();
047            parameters.addParameter(key, new Integer[] { page });
048        }
049    
050        public void setMaxRows(int maxRows) {
051            String key = prefixId + Action.MAX_ROWS.toParam();
052            parameters.addParameter(key, maxRows);
053        }
054    
055        public void addFilter(String property, String value) {
056            String key = prefixId + Action.FILTER.toParam() + property;
057            List<String> filterList = new ArrayList<String>();
058            filterList.add(value);
059            parameters.addParameter(key, filterList);
060        }
061    
062        /**
063         * Add a Sort parameter. Will increase the sort count.
064         * Note: do not mix and match this and the addSort()
065         * method that takes a position.
066         */
067        public void addSort(String property, Order order) {
068            addSort(++sortCount, property, order);
069        }
070        
071        public void addSort(int position, String property, Order order) {
072            String key = prefixId + Action.SORT.toParam() + position + "_" + property;
073            parameters.addParameter(key, new String[] { order.toParam() });
074        }
075    
076        public void setExportType(ExportType exportType) {
077            String key = prefixId + Action.EXPORT.toParam();
078            parameters.addParameter(key, exportType.toParam());
079        }
080        
081        public void setFilterWorksheet() {
082            String key = prefixId + FILTER_WORKSHEET;
083            parameters.addParameter(key, "true");
084        }
085    
086        public void setSaveWorksheet() {
087            String key = prefixId + SAVE_WORKSHEET;
088            parameters.addParameter(key, "true");
089            
090            String refreshKey = prefixId + TABLE_REFRESHING;
091            parameters.addParameter(refreshKey, "true");
092        }
093    }
