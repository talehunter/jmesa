016    package org.jmesa.view.component;
017    
018    import org.apache.commons.lang.StringUtils;
019    import org.jmesa.util.SupportUtils;
020    import org.jmesa.view.AbstractContextSupport;
021    import org.jmesa.view.ViewUtils;
022    import org.jmesa.view.editor.BasicCellEditor;
023    import org.jmesa.view.editor.CellEditor;
024    import org.jmesa.view.editor.FilterEditor;
025    import org.jmesa.view.editor.HeaderEditor;
026    import org.jmesa.view.renderer.BasicCellRenderer;
027    import org.jmesa.view.renderer.CellRenderer;
028    import org.jmesa.view.renderer.FilterRenderer;
029    import org.jmesa.view.renderer.HeaderRenderer;
030    
031    /**
032     * @since 2.0
033     * @author Jeff Johnston
034     */
035    public class Column extends AbstractContextSupport {
036        private String property;
037        private String title;
038        private String titleKey;
039    
040        private CellRenderer cellRenderer;
041        private CellEditor cellEditor;
042    
043        private HeaderRenderer headerRenderer;
044        private HeaderEditor headerEditor;
045    
046        private FilterRenderer filterRenderer;
047        private FilterEditor filterEditor;
048    
049        private Row row;
050    
051        public Column() {}
052    
053        public Column(String property) {
054            this.property = property;
055        }
056    
057        public String getProperty() {
058            return property;
059        }
060    
061        public void setProperty(String property) {
062            this.property = property;
063        }
064    
065        public Column property(String property) {
066            setProperty(property);
067            return this;
068        }
069        
070        public String getTitle() {
071            if (titleKey != null) {
072                return getCoreContext().getMessage(titleKey);
073            }
074    
075            if (StringUtils.isBlank(title)) {
076                return ViewUtils.camelCaseToWord(property);
077            }
078    
079            return title;
080        }
081    
082        public void setTitle(String title) {
083            this.title = title;
084        }
085    
086        public Column title(String title) {
087            setTitle(title);
088            return this;
089        }
090        
091        public void setTitleKey(String titleKey) {
092            this.titleKey = titleKey;
093        }
094    
095        public Column titleKey(String titleKey) {
096            setTitleKey(titleKey);
097            return this;
098        }
099        
100        public CellRenderer getCellRenderer() {
101            if (cellRenderer == null) {
102                this.cellRenderer = new BasicCellRenderer(this);
103            }
104            return cellRenderer;
105        }
106    
107        public void setCellRenderer(CellRenderer cellRenderer) {
108            this.cellRenderer = cellRenderer;
109            this.cellRenderer.setColumn(this);
110        }
111    
112        public Column cellRenderer(CellRenderer cellRenderer) {
113            setCellRenderer(cellRenderer);
114            return this;
115        }
116    
117        public CellEditor getCellEditor() {
118            if (cellEditor == null) {
119                this.cellEditor = new BasicCellEditor();
120            }
121            return cellEditor;
122        }
123    
124        public void setCellEditor(CellEditor cellEditor) {
125            this.cellEditor = cellEditor;
126    
127            /*
128             * This is useful for editors that are decorated at
129             * runtime. Most of the support classes are handled
130             * in the TableFacadeUtils.init() method though.
131             */
132            SupportUtils.setWebContext(cellEditor, getWebContext());
133            SupportUtils.setCoreContext(cellEditor, getCoreContext());
134            SupportUtils.setColumn(cellEditor, this);
135        }
136    
137        public Column cellEditor(CellEditor editor) {
138            setCellEditor(editor);
139            return this;
140        }
141    
142        public HeaderRenderer getHeaderRenderer() {
143            return headerRenderer;
144        }
145    
146        public void setHeaderRenderer(HeaderRenderer headerRenderer) {
147            this.headerRenderer = headerRenderer;
148            this.headerRenderer.setColumn(this);
149        }
150    
151        public Column headerRenderer(HeaderRenderer headerRenderer) {
152            setHeaderRenderer(headerRenderer);
153            return this;
154        }
155    
156        public HeaderEditor getHeaderEditor() {
157            return headerEditor;
158        }
159    
160        public void setHeaderEditor(HeaderEditor headerEditor) {
161            this.headerEditor = headerEditor;
162        }
163    
164        public Column headerEditor(HeaderEditor headerEditor) {
165            setHeaderEditor(headerEditor);
166            return this;
167        }
168    
169        public FilterRenderer getFilterRenderer() {
170            return filterRenderer;
171        }
172    
173        public void setFilterRenderer(FilterRenderer filterRenderer) {
174            this.filterRenderer = filterRenderer;
175            this.filterRenderer.setColumn(this);
176        }
177    
178        public Column filterRenderer(FilterRenderer filterRenderer) {
179            setFilterRenderer(filterRenderer);
180            return this;
181        }
182    
183        public FilterEditor getFilterEditor() {
184            return filterEditor;
185        }
186    
187        public void setFilterEditor(FilterEditor filterEditor) {
188            this.filterEditor = filterEditor;
189        }
190    
191        public Column filterEditor(FilterEditor filterEditor) {
192            setFilterEditor(filterEditor);
193            return this;
194        }
195    
196        public Row getRow() {
197            return row;
198        }
199    
200        public void setRow(Row row) {
201            this.row = row;
202        }
203    
204        public Column row(Row row) {
205            setRow(row);
206            return this;
207        }
208    }
