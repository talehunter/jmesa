016    package org.jmesa.view.html;
017    
018    import org.jmesa.core.CoreContext;
019    import org.jmesa.web.WebContext;
020    
021    /**
022     * @since 2.0
023     * @author Jeff Johnston
024     */
025    public class HtmlUtils {
026        private HtmlUtils() {}
027    
028        public static boolean isFirstPageEnabled(int page) {
029            if (page == 1) {
030                return false;
031            }
032    
033            return true;
034        }
035    
036        public static boolean isPrevPageEnabled(int page) {
037            if (page - 1 < 1) {
038                return false;
039            }
040    
041            return true;
042        }
043    
044        public static boolean isNextPageEnabled(int page, int totalPages) {
045            if (page + 1 > totalPages) {
046                return false;
047            }
048    
049            return true;
050        }
051    
052        public static boolean isLastPageEnabled(int page, int totalPages) {
053            if (page == totalPages || totalPages == 0) {
054                return false;
055            }
056    
057            return true;
058        }
059    
060        public static int totalPages(CoreContext coreContext) {
061            int maxRows = coreContext.getLimit().getRowSelect().getMaxRows();
062    
063            if (maxRows == 0) {
064                maxRows = coreContext.getLimit().getRowSelect().getTotalRows();
065            }
066    
067            int totalRows = coreContext.getLimit().getRowSelect().getTotalRows();
068    
069            int totalPages = 1;
070    
071            if (maxRows != 0) {
072                totalPages = totalRows / maxRows;
073            }
074    
075            if ((maxRows != 0) && ((totalRows % maxRows) > 0)) {
076                totalPages++;
077            }
078    
079            return totalPages;
080        }
081    
082        /**
083         * Find the starting rowcount for each page. The default is to start at 0. However, if the includePagination
084         * preference (html.rowcount.includePagination) is set to true then each page will start at the row
085         * that corresponds with the current page.
086         */
087        public static int startingRowcount(CoreContext coreContext) {
088            int rowcount = 0;
089    
090            boolean rowcountIncludePagination = Boolean.valueOf(coreContext.getPreference(HtmlConstants.ROWCOUNT_INCLUDE_PAGINATION));
091            if (rowcountIncludePagination) {
092                int page = coreContext.getLimit().getRowSelect().getPage();
093                int maxRows = coreContext.getLimit().getRowSelect().getMaxRows();
094                rowcount = (page - 1) * maxRows;
095            }
096    
097            return rowcount;
098        }
099    
100        /**
101         * Look in the preferences to find out if the document.ready script should be used.
102         * 
103         * @since 2.2
104         * @return Is true if including the document.ready script to initialize limit.
105         */
106        public static boolean useDocumentReadyToInitJavascriptLimit(CoreContext coreContext) {
107            String useDocumentReady = coreContext.getPreference(HtmlConstants.SNIPPETS_INIT_JAVASCRIPT_LIMIT_USE_DOCUMENT_READY);
108            return useDocumentReady.equals("true");
109        }
110    
111        public static String imagesPath(WebContext webContext, CoreContext coreContext) {
112            String imagesUrl = coreContext.getMessage(HtmlConstants.IMAGES_URL);
113            if (imagesUrl == null) {
114                imagesUrl = coreContext.getPreference(HtmlConstants.IMAGES_URL);
115            }
116            if (imagesUrl != null) {
117                return imagesUrl;
118            }
119    
120            String contextPath = webContext.getContextPath();
121            String imagesPath = coreContext.getMessage(HtmlConstants.IMAGES_PATH);
122            if (imagesPath == null) {
123                imagesPath = coreContext.getPreference(HtmlConstants.IMAGES_PATH);
124            }
125    
126            return contextPath + imagesPath;
127        }
128    }
