016    package org.jmesa.view.html.toolbar;
017    
018    import java.util.ArrayList;
019    import java.util.List;
020    import org.jmesa.limit.ExportType;
021    import org.jmesa.view.AbstractContextSupport;
022    import org.jmesa.view.html.HtmlBuilder;
023    import org.jmesa.limit.Limit;
024    import org.jmesa.limit.RowSelect;
025    import static org.jmesa.view.html.HtmlConstants.TOOLBAR_MAX_PAGE_NUMBERS;
026    import static org.jmesa.view.html.HtmlUtils.totalPages;
027    
028    /**
029     * The main logic to create toolbars.
030     *
031     * @since 2.2
032     * @author Jeff Johnston
033     */
034    public abstract class Toolbar extends AbstractContextSupport {
035        private ToolbarItemFactory toolbarItemFactory;
036        private List<ToolbarItem> toolbarItems = new ArrayList<ToolbarItem>();
037    
038        private ToolbarItemFactory getToolbarItemFactory() {
039            if (toolbarItemFactory == null) {
040                this.toolbarItemFactory = new ToolbarItemFactory(getWebContext(), getCoreContext());
041            }
042    
043            return toolbarItemFactory;
044        }
045    
046        protected boolean hasToolbarItems() {
047            return toolbarItems != null && toolbarItems.size() > 0;
048        }
049    
050        public List<ToolbarItem> getToolbarItems() {
051            return toolbarItems;
052        }
053    
054        public void addToolbarItem(ToolbarItem item) {
055            toolbarItems.add(item);
056        }
057    
058        public ToolbarItem addToolbarItem(ToolbarItemType type) {
059            ToolbarItem item = null;
060    
061            ToolbarItemFactory factory = getToolbarItemFactory();
062    
063            switch (type) {
064                case PAGE_NUMBER_ITEMS:
065                    addPageNumberItems();
066                    break;
067                case FIRST_PAGE_ITEM:
068                    item = factory.createFirstPageItem();
069                    break;
070                case PREV_PAGE_ITEM:
071                    item = factory.createPrevPageItem();
072                    break;
073                case NEXT_PAGE_ITEM:
074                    item = factory.createNextPageItem();
075                    break;
076                case LAST_PAGE_ITEM:
077                    item = factory.createLastPageItem();
078                    break;
079                case MAX_ROWS_ITEM:
080                    item = factory.createMaxRowsItem();
081                    break;
082                case FILTER_ITEM:
083                    item = factory.createFilterItem();
084                    break;
085                case CLEAR_ITEM:
086                    item = factory.createClearItem();
087                    break;
088                case SEPARATOR:
089                    item = factory.createSeparatorItem();
090                    break;
091                case SAVE_WORKSHEET_ITEM:
092                    item = factory.createSaveWorksheetItem();
093                    break;
094                case FILTER_WORKSHEET_ITEM:
095                    item = factory.createFilterWorksheetItem();
096                    break;
097                case CLEAR_WORKSHEET_ITEM:
098                    item = factory.createClearWorksheetItem();
099                    break;
100                case ADD_WORKSHEET_ROW_ITEM:
101                    item = factory.createAddWorksheetRowItem();
102                    break;
103            }
104    
105            if (item != null) {
106                toolbarItems.add(item);
107            }
108    
109            return item;
110        }
111    
112        public List<ToolbarItem> addExportToolbarItems(ExportType... exportTypes) {
113            List<ToolbarItem> items = new ArrayList<ToolbarItem>();
114    
115            if (exportTypes == null || exportTypes.length == 0) {
116                return items;
117            }
118    
119            for (int i = 0; i < exportTypes.length; i++) {
120                ExportType exportType = exportTypes[i];
121                items.add(addExportToolbarItem(exportType));
122            }
123    
124            return items;
125        }
126    
127        public ToolbarItem addExportToolbarItem(ExportType exportType) {
128            ToolbarExport export = new ToolbarExport(exportType);
129            ToolbarItemFactory factory = getToolbarItemFactory();
130            ToolbarItem item = factory.createExportItem(export);
131            toolbarItems.add(item);
132            return item;
133        }
134    
135        private void addPageNumberItems() {
136            ToolbarItemFactory factory = getToolbarItemFactory();
137    
138            Limit limit = getCoreContext().getLimit();
139            RowSelect rowSelect = limit.getRowSelect();
140            int page = rowSelect.getPage();
141            int totalPages = totalPages(getCoreContext());
142    
143            int maxPages = Integer.valueOf(getCoreContext().getPreference(TOOLBAR_MAX_PAGE_NUMBERS));
144    
145            int centerPage = maxPages/2 + 1;
146            int startEndPages = maxPages/2;
147    
148            if (totalPages > maxPages) {
149                int start;
150                int end;
151    
152                if (page <= centerPage) { // the start of the pages
153                    start = 1;
154                    end = maxPages;
155                } else if (page >= totalPages - startEndPages) { // the last few pages
156                    start = totalPages - (maxPages - 1);
157                    end = totalPages;
158                } else { // center everything else
159                    start = page - startEndPages;
160                    end = page + startEndPages;
161                }
162    
163                for (int i = start; i <= end; i++) {
164                    addToolbarItem(factory.createPageNumberItem(i));
165                }
166            } else {
167                for (int i = 1; i <= totalPages; i++) {
168                    addToolbarItem(factory.createPageNumberItem(i));
169                }
170            }
171        }
172    
173        public String render() {
174            HtmlBuilder html = new HtmlBuilder();
175    
176            html.table(2).border("0").cellpadding("0").cellspacing("1").close();
177    
178            html.tr(3).close();
179    
180            for (ToolbarItem item : toolbarItems) {
181                html.td(4).close();
182                html.append(item.getToolbarItemRenderer().render());
183                html.tdEnd();
184            }
185    
186            html.trEnd(3);
187    
188            html.tableEnd(2);
189            html.newline();
190            html.tabs(2);
191    
192            return html.toString();
193        }
194    }
