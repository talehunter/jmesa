046    /**
047     * A PDF view that uses the iText PdfPTable.
048     *
049     * @since 2.3.4
050     * @author Ismail Seyfi
051     */
052    public class PdfPView extends AbstractExportView {
053        private Logger logger = LoggerFactory.getLogger(PdfPView.class);
054    
055        private Color evenCellBackgroundColor;
056        private Color oddCellBackgroundColor;
057        private Color headerBackgroundColor;
058        private Color headerFontColor;
059        private Color captionFontColor;
060        private String captionAlignment;
061    
062        public PdfPView() {
063            this.evenCellBackgroundColor = new Color(227, 227, 227);
064            this.oddCellBackgroundColor = new Color(255, 255, 255);
065            this.headerBackgroundColor = new Color(114, 159, 207);
066            this.headerFontColor = new Color(255, 255, 255);
067            this.captionFontColor = new Color(0, 0, 0);
068            this.captionAlignment = "center";
069        }
070    
071        public Paragraph getTableCaption() throws Exception {
072            Paragraph p = new Paragraph(getTable().getCaption(), getFont(HELVETICA, 18, BOLD, getCaptionFontColor()));
073            p.setAlignment(getCaptionAlignment());
074            return p;
075        }
076    
077        public PdfPTable render() {
078            PdfPTable pdfpTable = new PdfPTable(getTable().getRow().getColumns().size());
079            pdfpTable.setSpacingBefore(3);
080    
081            Row row = getTable().getRow();
082    
083            List<Column> columns = row.getColumns();
084    
085            // build table headers
086            for (Iterator<Column> iter = columns.iterator(); iter.hasNext();) {
087                Column column = iter.next();
088                PdfPCell cell = new PdfPCell(new Paragraph(column.getTitle(), getHeaderCellFont()));
089                cell.setPadding(3.0f);
090                cell.setBackgroundColor(getHeaderBackgroundColor());
091                pdfpTable.addCell(cell);
092            }
093    
094            // build table body
095            Collection<?> items = getCoreContext().getPageItems();
096            int rowcount = 0;
097            for (Object item : items) {
098                rowcount++;
099    
100                columns = row.getColumns();
101    
102                for (Iterator<Column> iter = columns.iterator(); iter.hasNext();) {
103                    Column column = iter.next();
104    
105                    String property = column.getProperty();
106                    Object value = column.getCellRenderer().getCellEditor().getValue(item, property, rowcount);
107                    PdfPCell cell = new PdfPCell(new Paragraph(value == null ? "" : String.valueOf(value), getCellFont()));
108                    cell.setPadding(3.0f);
109    
110                    if (isRowEven(rowcount)) {
111                        cell.setBackgroundColor(getEvenCellBackgroundColor());
112                    } else {
113                        cell.setBackgroundColor(getOddCellBackgroundColor());
114                    }
115    
116                    pdfpTable.addCell(cell);
117                }
118            }
119    
120            return pdfpTable;
121        }

147        /**
148         * Create either the default helvetica 12 point font, or specify the
149         * font name and encoding in the preferences. Either way it will use
150         * the header font color.

154         *  export.pdf.fontName
155         *  export.pdf.fontEncoding
156         * </p>

176         *  export.pdf.fontName
177         *  export.pdf.fontEncoding
178         * </p>

180        public Font getCellFont() {
181            return getFontWithColor(null);

199    
200        private Font getFontWithColor(Color color) {
201            String fontName = getCoreContext().getPreference(PDF_FONT_NAME);
202            String fontEncoding = getCoreContext().getPreference(PDF_FONT_ENCODING);
203            if (isNotBlank(fontName) && isNotBlank(fontEncoding)) {
204                try {
205                    BaseFont baseFont = createFont(fontName, fontEncoding, NOT_EMBEDDED);
206                    if (color != null) {
207                        return new Font(baseFont, 12, 0, color);
208                    }
209                    return new Font(baseFont, 12, 0);
210                } catch (Exception e) {
211                    logger.warn("Not able to create the requested font for the PDF export...will use the export.");
212                }
213            }
214    
215            if (color != null) {
216                return getFont(HELVETICA, 12, NORMAL, color);
217            }
218    
219            return getFont(HELVETICA, 12, NORMAL);
220        }
