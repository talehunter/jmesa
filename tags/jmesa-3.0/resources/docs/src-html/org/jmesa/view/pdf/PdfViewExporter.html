016    package org.jmesa.view.pdf;
017    
018    import java.io.ByteArrayInputStream;
019    import java.io.UnsupportedEncodingException;
020    
021    import javax.servlet.http.HttpServletRequest;
022    import javax.servlet.http.HttpServletResponse;
023    import javax.xml.parsers.DocumentBuilder;
024    import javax.xml.parsers.DocumentBuilderFactory;
025    
026    import org.jmesa.core.CoreContext;
027    import org.jmesa.view.AbstractViewExporter;
028    import org.jmesa.view.View;
029    import org.slf4j.Logger;
030    import org.slf4j.LoggerFactory;
031    import org.w3c.dom.Document;
032    import org.xhtmlrenderer.pdf.ITextRenderer;
033    import org.xhtmlrenderer.util.XRLog;
034    import org.xhtmlrenderer.resource.FSEntityResolver;
035    
036    /**
037     * @since 2.2
038     * @author Paul Horn
039     */
040    public class PdfViewExporter extends AbstractViewExporter {
041        private static Logger logger = LoggerFactory.getLogger(PdfViewExporter.class);
042        private HttpServletRequest request;
043    
044        public PdfViewExporter(View view, CoreContext coreContext, HttpServletRequest request, HttpServletResponse response) {
045            super(view, coreContext, response);
046            this.request = request;
047        }
048    
049        public PdfViewExporter(View view, CoreContext coreContext, HttpServletRequest request, HttpServletResponse response, String fileName) {
050            super(view, coreContext, response, fileName);
051            this.request = request;
052        }
053    
054        public void export() throws Exception {
055            String string = (String) getView().render();
056    
057            byte[] contents = null;
058            try {
059                contents = string.getBytes("UTF-8");
060            } catch (UnsupportedEncodingException e) {
061                logger.info("Not able to process the PDF file using the UTF-8 encoding.");
062            }
063    
064            HttpServletResponse response = getResponse();
065            responseHeaders(response);
066    
067            System.setProperty("xr.util-logging.loggingEnabled", "false");
068            System.setProperty("xr.util-logging.java.util.logging.ConsoleHandler.level", "WARN");
069            System.setProperty("xr.util-logging..level", "WARN");
070            System.setProperty("xr.util-logging.plumbing.level", "WARN");
071            System.setProperty("xr.util-logging.plumbing.config.level", "WARN");
072            System.setProperty("xr.util-logging.plumbing.exception.level", "WARN");
073            System.setProperty("xr.util-logging.plumbing.general.level", "WARN");
074            System.setProperty("xr.util-logging.plumbing.init.level", "WARN");
075            System.setProperty("xr.util-logging.plumbing.load.level", "WARN");
076            System.setProperty("xr.util-logging.plumbing.load.xml-entities.level", "WARN");
077            System.setProperty("xr.util-logging.plumbing.match.level", "WARN");
078            System.setProperty("xr.util-logging.plumbing.cascade.level", "WARN");
079            System.setProperty("xr.util-logging.plumbing.css-parse.level", "WARN");
080            System.setProperty("xr.util-logging.plumbing.layout.level", "WARN");
081            System.setProperty("xr.util-logging.plumbing.render.level", "WARN");
082    
083            XRLog.setLoggingEnabled(false);
084    
085            ITextRenderer renderer = new ITextRenderer();
086    
087            DocumentBuilder builder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
088            FSEntityResolver er= FSEntityResolver.instance();
089            builder.setEntityResolver(er);
090            Document doc = builder.parse(new ByteArrayInputStream(contents));
091    
092            renderer.setDocument(doc, getBaseUrl());
093            renderer.layout();
094            renderer.createPDF(response.getOutputStream());
095        }
096    
097        /**
098         * @return The base url to the web application.
099         */
100        private String getBaseUrl() {
101            if (request != null) {
102                return request.getRequestURL().toString();
103            }
104            return null;
105        }
106    
107        public String getContextType() {
108            return "application/pdf";
109        }
110    
111        public String getExtensionName() {
112            return "pdf";
113        }
114    }
