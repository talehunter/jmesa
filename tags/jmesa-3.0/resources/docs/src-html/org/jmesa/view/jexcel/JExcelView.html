016    package org.jmesa.view.jexcel;
017    
018    import java.io.OutputStream;
019    import java.util.Collection;
020    import java.util.List;
021    
022    import jxl.Workbook;
023    import jxl.format.Border;
024    import jxl.format.BorderLineStyle;
025    import jxl.format.Colour;
026    import jxl.write.Label;
027    import jxl.write.Number;
028    import jxl.write.WritableCell;
029    import jxl.write.WritableCellFormat;
030    import jxl.write.WritableFont;
031    import jxl.write.WritableSheet;
032    import jxl.write.WritableWorkbook;
033    
034    import org.apache.commons.lang.StringUtils;
035    import org.jmesa.view.AbstractExportView;
036    import org.jmesa.view.component.Column;
037    import org.jmesa.view.component.Row;
038    import org.jmesa.view.component.Table;
039    
040    /**
041     * <p>
042     * Create the JExcel view.
043     * </p>
044     *
045     * @since 2.2
046     * @author Paul Horn
047     */
048    public class JExcelView extends AbstractExportView {
049        private OutputStream outputStream;
050    
051        public void setOutputStream(OutputStream outputStream) {
052            this.outputStream = outputStream;
053        }
054    
055        public Object render() {
056            try {
057                Table table = getTable();
058                WritableWorkbook workbook = Workbook.createWorkbook(outputStream);
059                String caption = table.getCaption();
060                if (StringUtils.isEmpty(caption)) {
061                    caption = "JMesa Export";
062                }
063                workbook.createSheet(caption, 0);
064    
065                WritableCellFormat headerFmt = new WritableCellFormat(new WritableFont(WritableFont.ARIAL, 10));
066                headerFmt.setBackground(Colour.GREY_25_PERCENT);
067                headerFmt.setBorder(Border.ALL, BorderLineStyle.THIN);
068    
069                WritableSheet sheet = workbook.getSheet(caption);
070    
071                int colidx = 0;
072                int rowidx = 0;
073    
074                Row row = table.getRow();
075                row.getRowRenderer();
076                List<Column> columns = table.getRow().getColumns();
077    
078                for (Column col : columns) {
079                    sheet.addCell(new Label(colidx, rowidx, col.getTitle(), headerFmt));
080                    sheet.setColumnView(colidx++, 20);
081                }
082    
083                rowidx++;
084    
085                WritableCellFormat rowFmt = new WritableCellFormat(new WritableFont(WritableFont.ARIAL, 10));
086    
087                Collection<?> items = getCoreContext().getPageItems();
088                for (Object item : items) {
089                    colidx = 0;
090                    for (Column col : columns) {
091                        Object value = col.getCellRenderer().render(item, rowidx);
092                        if (value == null) {
093                            value = "";
094                        }
095    
096                        WritableCell cell = null;
097                        if (value instanceof java.lang.Number) {
098                            cell = new Number(colidx++, rowidx, Double.valueOf(value.toString()), rowFmt);
099                        } else {
100                            cell = new Label(colidx++, rowidx, value + "", rowFmt);
101                        }
102    
103                        sheet.addCell(cell);
104                    }
105                    rowidx++;
106                }
107    
108                return workbook;
109            } catch (Exception e) {
110                throw new RuntimeException(e);
111            }
112        }
113    }
