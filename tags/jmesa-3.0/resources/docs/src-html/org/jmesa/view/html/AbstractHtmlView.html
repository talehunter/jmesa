016    package org.jmesa.view.html;
017    
018    import org.jmesa.util.SupportUtils;
019    import org.jmesa.view.AbstractContextSupport;
020    import org.jmesa.view.View;
021    import org.jmesa.view.component.Table;
022    import org.jmesa.view.component.TableSupport;
023    import org.jmesa.view.html.component.HtmlTable;
024    import org.jmesa.view.html.toolbar.Toolbar;
025    import org.jmesa.view.html.toolbar.ToolbarSupport;
026    
027    /**
028     * Abstract view for building html tables.
029     * 
030     * @since 2.2
031     * @author Jeff Johnston
032     */
033    public abstract class AbstractHtmlView extends AbstractContextSupport implements View, ToolbarSupport, TableSupport {
034        private HtmlTable table;
035        private Toolbar toolbar;
036        private HtmlSnippets snippets;
037    
038        public HtmlTable getTable() {
039            return table;
040        }
041    
042        public void setTable(Table table) {
043            this.table = (HtmlTable) table;
044        }
045    
046        public Toolbar getToolbar() {
047            return toolbar;
048        }
049    
050        public void setToolbar(Toolbar toolbar) {
051            this.toolbar = toolbar;
052        }
053    
054        public byte[] getBytes() {
055            String render = (String) render();
056            return render.getBytes();
057        }
058    
059        /**
060         * Return the HtmlSnippets object. If the snippets is null then the default one will be created.
061         * 
062         * @return The HtmlSnippets object.
063         */
064        public HtmlSnippets getHtmlSnippets() {
065            if (snippets == null) {
066                this.snippets = new HtmlSnippetsImpl(getTable(), getToolbar(), getCoreContext());
067            }
068            SupportUtils.setWebContext(this.snippets, getWebContext());
069            return snippets;
070        }
071    
072        public void setHtmlSnippets(HtmlSnippets snippets) {
073            this.snippets = snippets;
074        }
075    }
