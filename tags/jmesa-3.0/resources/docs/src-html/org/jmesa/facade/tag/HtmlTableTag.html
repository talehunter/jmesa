016    package org.jmesa.facade.tag;
017    
018    import static org.jmesa.facade.tag.TagUtils.getTableTableRenderer;
019    
020    import java.io.IOException;
021    
022    import javax.servlet.jsp.JspException;
023    import javax.servlet.jsp.tagext.JspFragment;
024    import javax.servlet.jsp.tagext.SimpleTagSupport;
025    
026    import org.jmesa.view.html.HtmlComponentFactory;
027    import org.jmesa.view.html.component.HtmlTable;
028    import org.jmesa.view.html.renderer.HtmlTableRenderer;
029    
030    /**
031     * Represents an HtmlTable.
032     * 
033     * @since 2.1
034     * @author jeff jie
035     */
036    public class HtmlTableTag extends SimpleTagSupport {
037        // core attributes
038        private String caption;
039        private String captionKey;
040    
041        // style attributes
042        private String theme;
043        private String tableRenderer;
044        private String style;
045        private String styleClass;
046        private String width;
047        private String border;
048        private String cellpadding;
049        private String cellspacing;
050    
051        public String getCaption() {
052            return caption;
053        }
054    
055        public void setCaption(String caption) {
056            this.caption = caption;
057        }
058    
059        public String getCaptionKey() {
060            return captionKey;
061        }
062    
063        public void setCaptionKey(String captionKey) {
064            this.captionKey = captionKey;
065        }
066    
067        public String getTheme() {
068            return theme;
069        }
070    
071        public void setTheme(String theme) {
072            this.theme = theme;
073        }
074    
075        /**
076         * @since 2.2
077         */
078        public String getTableRenderer() {
079            return tableRenderer;
080        }
081    
082        /**
083         * @since 2.2
084         */
085        public void setTableRenderer(String tableRenderer) {
086            this.tableRenderer = tableRenderer;
087        }
088    
089        public String getStyle() {
090            return style;
091        }
092    
093        public void setStyle(String style) {
094            this.style = style;
095        }
096    
097        public String getStyleClass() {
098            return styleClass;
099        }
100    
101        public void setStyleClass(String styleClass) {
102            this.styleClass = styleClass;
103        }
104    
105        public String getWidth() {
106            return width;
107        }
108    
109        public void setWidth(String width) {
110            this.width = width;
111        }
112    
113        public String getBorder() {
114            return border;
115        }
116    
117        public void setBorder(String border) {
118            this.border = border;
119        }
120    
121        public String getCellpadding() {
122            return cellpadding;
123        }
124    
125        public void setCellpadding(String cellpadding) {
126            this.cellpadding = cellpadding;
127        }
128    
129        public String getCellspacing() {
130            return cellspacing;
131        }
132    
133        public void setCellspacing(String cellspacing) {
134            this.cellspacing = cellspacing;
135        }
136    
137        /**
138         * The table to use. If the table does not exist then one will be created.
139         */
140        private HtmlTable getTable(HtmlComponentFactory factory) {
141            HtmlTable table = factory.createTable();
142            table.setCaption(getCaption());
143            table.setCaptionKey(getCaptionKey());
144            table.setTheme(getTheme());
145    
146            HtmlTableRenderer renderer = getTableTableRenderer(table, getTableRenderer());
147            table.setTableRenderer(renderer);
148    
149            renderer.setWidth(getWidth());
150            renderer.setStyle(getStyle());
151            renderer.setStyleClass(getStyleClass());
152            renderer.setBorder(getBorder());
153            renderer.setCellpadding(getCellpadding());
154            renderer.setCellspacing(getCellspacing());
155    
156            return table;
157        }
158    
159        @Override
160        public void doTag() throws JspException, IOException {
161            JspFragment body = getJspBody();
162            if (body == null) {
163                throw new IllegalStateException("You need to wrap the table in the facade tag.");
164            }
165    
166            TableFacadeTag facadeTag = (TableFacadeTag) findAncestorWithClass(this, TableFacadeTag.class);
167            HtmlTable table = facadeTag.getTable();
168            if (table == null) {
169                HtmlComponentFactory factory = facadeTag.getComponentFactory();
170                table = getTable(factory);
171                facadeTag.setTable(table);
172            }
173    
174            body.invoke(null);
175        }
176    }
