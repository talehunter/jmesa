061        public static final String TABLE_REFRESHING = "tr_";
062        public static final String CLEARING_WORKSHEET = "cw_";
063        
064        private TableFacadeUtils() {}

066        /**
067         * There needs to be a way to tell if the table is refreshing, as opposed to being run for the first time.
068         * 
069         * @param id The table identifier.
070         * @param webContext The web context.
071         * @return Is true if the table is being refreshed.
072         */
073        static boolean isTableRefreshing(String id, WebContext webContext) {
074            String refreshing = webContext.getParameter(id + "_" + TABLE_REFRESHING);
075            if (StringUtils.isNotEmpty(refreshing) && refreshing.equals("true")) {
076                return true;
077            }
078            
079            return false;
080        }
081        
082        /**
083         * @param id The table identifier.
084         * @param webContext The web context.
085         * @return Is true if the user is requesting to clear all the worksheet changes.
086         */
087        static boolean isClearingWorksheet(String id, WebContext webContext) {
088            String clearingWorksheet = webContext.getParameter(id + "_" + CLEARING_WORKSHEET);
089            if (StringUtils.isNotEmpty(clearingWorksheet) && clearingWorksheet.equals("true")) {
090                return true;
091            }
092            
093            return false;
094        }

096        /**
097         * Filter the items by the rows in the worksheet.
098         * 
099         * @param items The collection of beans or maps.
100         * @param worksheet The current worksheet.
101         * @return The filtered items.
102         */
103        static Collection<?> filterWorksheetItems(Collection<?> items, Worksheet worksheet) {
104            if (!worksheet.isFiltering()) {
105                return items;
106            }
107    
108            Collection<WorksheetRow> worksheetRows = worksheet.getRows();
109    
110            if (items.size() == worksheetRows.size()) {
111                return items;
112            }
113    
114            List<Object> results = new ArrayList<Object>();
115    
116            for (Object item : items) {
117                for (WorksheetRow worksheetRow : worksheetRows) {
118                    String uniqueProperty = worksheetRow.getUniqueProperty().getName();
119                    String uniquePropertyValue = worksheetRow.getUniqueProperty().getValue();
120                    try {
121                        Object value = PropertyUtils.getProperty(item, uniqueProperty);
122                        if (value.toString().equals(uniquePropertyValue)) {
123                            results.add(item);
124                        }
125    
126                    } catch (Exception e) {
127                        logger.error("Had problems evaluating the items.", e);
128                    }
129                }
130            }
131    
132            return results;
133        }

155        /**
156         * Spin through the components and inject the proper support classes. This is kind
157         * of a mini IOC implementation.
158         */
159        public static void initTable(TableFacade tableFacade, Table table) {
160            WebContext webContext = tableFacade.getWebContext();
161            CoreContext coreContext = tableFacade.getCoreContext();
162    
163            // get the table set up
164    
165            init(table, webContext, coreContext);
166    
167            TableRenderer tableRenderer = table.getTableRenderer();
168            init(tableRenderer, webContext, coreContext);
169    
170            // get the row set up
171    
172            Row row = table.getRow();
173            init(row, webContext, coreContext);
174    
175            if (row instanceof HtmlRow) {
176                HtmlRow htmlRow = (HtmlRow)row;
177                init(htmlRow.getOnclick(), webContext, coreContext);
178                init(htmlRow.getOnmouseover(), webContext, coreContext);
179                init(htmlRow.getOnmouseout(), webContext, coreContext);
180            }
181    
182            RowRenderer rowRenderer = row.getRowRenderer();
183            init(rowRenderer, webContext, coreContext);
184    
185            // get the column set up
186    
187            for (Column column : row.getColumns()) {
188                init(column, webContext, coreContext);
189    
190                // cell
191    
192                CellRenderer cellRenderer = column.getCellRenderer();
193                init(cellRenderer, webContext, coreContext);
194                SupportUtils.setColumn(cellRenderer, column);
195    
196                CellEditor cellEditor = column.getCellEditor();
197                init(cellEditor, webContext, coreContext);
198                SupportUtils.setColumn(cellEditor, column);
199    
200                // header
201    
202                HeaderRenderer headerRenderer = column.getHeaderRenderer();
203                init(headerRenderer, webContext, coreContext);
204                SupportUtils.setColumn(headerRenderer, column);
205    
206                HeaderEditor headerEditor = column.getHeaderEditor();
207                init(headerEditor, webContext, coreContext);
208                SupportUtils.setColumn(headerEditor, column);
209    
210                // filter
211    
212                if (column instanceof HtmlColumn) {
213                    HtmlColumn htmlColumn = (HtmlColumn)column;
214                    
215                    WorksheetEditor worksheetEditor = htmlColumn.getWorksheetEditor();
216                    if (worksheetEditor != null) {
217                        init(worksheetEditor, webContext, coreContext);
218                        SupportUtils.setColumn(worksheetEditor, column);
219    
220                        CellEditor worksheetCellEditor = worksheetEditor.getCellEditor();
221                        init(worksheetCellEditor, webContext, coreContext);
222                        SupportUtils.setColumn(worksheetCellEditor, column);
223                    }
224    
225                    FilterRenderer filterRenderer = htmlColumn.getFilterRenderer();
226                    init(filterRenderer, webContext, coreContext);
227                    SupportUtils.setColumn(filterRenderer, column);
228    
229                    FilterEditor filterEditor = column.getFilterEditor();
230                    init(filterEditor, webContext, coreContext);
231                    SupportUtils.setColumn(filterEditor, column);
232                }
233            }
234        }
235    
236        private static void init(Object obj, WebContext webContext, CoreContext coreContext) {
237            SupportUtils.setWebContext(obj, webContext);
238            SupportUtils.setCoreContext(obj, coreContext);
239        }
