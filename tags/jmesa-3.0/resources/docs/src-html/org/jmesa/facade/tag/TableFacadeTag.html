453         * These are the converted items. What happens is the regular items are passed to the columns,
454         * but then the columns are resolved with the EL Expression Language. A new Map is swapped out
455         * for each row Bean (or Map) and then used in place of the regular items. So in effect this is
456         * a Collection of Maps that represents the original Collection of Beans (or Maps) after being
457         * processed by the columns.
458         * 
459         * @return The converted items.
460         */

462            return pageItems;

465        /**
466         * Process the table, rows, and columns. Will pass the current bean in the pageScope for each
467         * bean in the Collection of Beans or Collection of Maps.
468         */
469        @Override
470        public void doTag() throws JspException, IOException {
471            JspFragment body = getJspBody();
472            if (body == null) {
473                throw new IllegalStateException("You need to wrap the table in the facade tag.");
474            }
475    
476            this.tableFacade = createTableFacade();
477            tableFacade.setPreferences(getTableFacadePreferences(getPreferences()));
478            tableFacade.setEditable(isEditable());
479            tableFacade.setItems(getItems());
480            tableFacade.setMaxRows(getMaxRows());
481            tableFacade.setStateAttr(getStateAttr());
482            tableFacade.setState(getTableFacadeState(getState()));
483            tableFacade.setLimit(getLimit());
484    
485            tableFacade.setMaxRowsIncrements(getTableFacadeMaxRowIncrements(getMaxRowsIncrements()));
486            tableFacade.setExportTypes(getTableFacadeExportTypes(getExportTypes()));
487    
488            tableFacade.autoFilterAndSort(isAutoFilterAndSort());
489            tableFacade.setColumnSort(getTableFacadeColumnSort(getColumnSort()));
490            tableFacade.setRowFilter(getTableFacadeRowFilter(getRowFilter()));
491            tableFacade.addFilterMatcherMap(getTableFacadeFilterMatcherMap(getFilterMatcherMap()));
492    
493            Collection<?> pi = tableFacade.getCoreContext().getPageItems();
494    
495            if (pi.size() == 0) {
496                body.invoke(null);
497                getPageItems().clear();
498            } else {
499                int count = HtmlUtils.startingRowcount(tableFacade.getCoreContext());
500                for (Iterator<?> iterator = pi.iterator(); iterator.hasNext();) {
501                    Object item = iterator.next();
502                    getJspContext().setAttribute(getVar(), item);
503                    getJspContext().setAttribute("rowcount", ++count);
504                    body.invoke(null);
505                }
506            }
507    
508            getJspContext().removeAttribute(getVar()); // clean up the page scoped bean
509            getJspContext().removeAttribute("rowcount");
510    
511            tableFacade.setTable(getTable());
512            tableFacade.setToolbar(getTableFacadeToolbar(getToolbar()));
513            tableFacade.setView(getTableFacadeView(getView()));
514            tableFacade.getCoreContext().setPageItems(getPageItems());
515    
516            Worksheet worksheet = tableFacade.getWorksheet();
517            if (ViewUtils.isEditable(worksheet)) {
518                    if (worksheet.isAddingRow()) {
519                            tableFacade.addWorksheetRow(getAddedRowObject());
520                    }
521                    
522                    if (worksheet.isRemovingRow()) {
523                            tableFacade.removeWorksheetRow();
524                    }
525            }
526    
527            View v = tableFacade.getView();
528            String html = v.render().toString();
529            getJspContext().getOut().print(html);
530        }

532        protected WebContext getWebContext() {
533            return new JspPageWebContext((PageContext) getJspContext());
534        }

536        protected TableFacade createTableFacade() {
537    
538            TableFacade facade = TableFacadeFactory.createTableFacade(getId(), getWebContext());
539            facade.setMessages(getTableFacadeMessages(getMessages()));
540            return facade;
