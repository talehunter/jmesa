016    package org.jmesa.util;
017    
018    import java.lang.reflect.Method;
019    
020    import org.apache.log4j.Logger;
021    
022    /**
023     * Utilities for java bean.
024     *
025     * @author wliu
026     */
027    public class BeanUtils {
028        private static final Logger logger = Logger.getLogger(BeanUtils.class);
029        
030        /**
031         * the possible prefixes for read method
032         */
033        private static final String[] getterPrefixes = { "get", "is"};
034        
035        /**
036         * Gets the type of a property of an object.
037         * 
038         * @param object The object that the property belongs to, cannot be null.
039         * @param property The property to get type, can be nested. for example, 'foo.bar.baz'.
040         * @return The type of the property. If the property doesn't exists in the object, returns null.
041         */
042        public static Class getPropertyType(Object object, String property) {
043            if (object == null) {
044                throw new IllegalArgumentException("Object cannot be null.");
045            }
046            return getPropertyType(object.getClass(), property);
047        }
048    
049        /**
050         * Gets the type of a property of a class.
051         * 
052         * @param clazz The class that the property belongs to, cannot be null.
053         * @param property The property to get type, can be nested. for example, 'foo.bar.baz'.
054         * @return The type of the property. If the property doesn't exists in the clazz, returns null.
055         */
056        public static Class getPropertyType(Class clazz, String property) {
057            if (clazz == null) {
058                throw new IllegalArgumentException("Clazz cannot be null.");
059            }
060    
061            if (property == null) {
062                throw new IllegalArgumentException("Property cannot be null.");
063            }
064    
065            int dotIndex = property.lastIndexOf('.');
066    
067            if (dotIndex == -1) {
068                Method method = getReadMethod(clazz, property);
069                return method == null ? null : method.getReturnType();
070            } 
071                
072            String deepestProperty = property.substring(dotIndex + 1);
073            String parentProperty = property.substring(0, dotIndex);
074            return getPropertyType(getPropertyType(clazz, parentProperty), deepestProperty);
075        }
076        
077        /**
078         * Gets the read method for a property in a class.
079         * 
080         * for example: <code>
081         * class Foo {
082         *      public String getBar() { return "bar"; } 
083         *      public Boolean isBaz() { return false; }
084         * }
085         * 
086         * BeanUtils.getReadMethod(Foo.class, "bar"); // return Foo#getBar()
087         * BeanUtils.getReadMethod(Foo.class, "baz"); // return Foo#isBaz()
088         * BeanUtils.getReadMethod(Foo.class, "baa"); // return null
089         * </code>
090         * 
091         * @param clazz The class to get read method.
092         * @param property The property to get read method for, can NOT be nested.
093         * @return The read method (getter) for the property, if there is no read method for
094         *          the property, returns null.
095         */
096        public static Method getReadMethod(Class clazz, String property) {
097            // Capitalize the property 
098            StringBuffer buf = new StringBuffer();
099            buf.append(property.substring(0, 1).toUpperCase());
100            if (property.length() > 1) {
101                buf.append(property.substring(1));
102            }
103            
104            Method method = null;
105            for (String prefix : getterPrefixes) {
106                String methodName = prefix + buf.toString();
107                try {
108                    method = clazz.getMethod(methodName);
109                    
110                    // Once get method successfully, jump out the loop. 
111                    break;
112                } catch (NoSuchMethodException e) {
113                    // do nothing but logging
114                    logger.debug("No such read method '" + methodName + "()' in class '" +
115                            clazz.getName() + "'.");
116                } catch (SecurityException e) {
117                    // do nothing but logging
118                    logger.debug("Error occurs while getting read method '" + methodName + "()' in class '" +
119                            clazz.getName() + "'.");
120                }
121            }
122            
123            return method;
124        }
125    }
