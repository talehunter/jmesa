016    package org.jmesa.core.filter;
017    
018    import org.jmesa.view.editor.*;
019    import org.jmesa.web.WebContext;
020    import org.jmesa.web.WebContextSupport;
021    
022    /**
023     * Abstract class to hold the pattern information for filter matcher classes.
024     * 
025     * @since 2.4.4
026     * @author Jeff Johnston
027     */
028    public abstract class AbstractPatternFilterMatcher implements FilterMatcher, PatternSupport, WebContextSupport {
029        private String pattern;
030        private WebContext webContext;
031    
032        public String getPattern() {
033            return pattern;
034        }
035    
036        public void setPattern(String pattern) {
037            this.pattern = pattern;
038        }
039    
040        public WebContext getWebContext() {
041            return webContext;
042        }
043    
044        public void setWebContext(WebContext webContext) {
045            this.webContext = webContext;
046        }
047    }
