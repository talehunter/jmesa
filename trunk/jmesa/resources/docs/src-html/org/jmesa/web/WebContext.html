016    package org.jmesa.web;
017    
018    import java.io.Writer;
019    import java.util.Locale;
020    import java.util.Map;
021    
022    /**
023     * @since 2.0
024     * @author Jeff Johnston
025     */
026    public interface WebContext {
027        public Object getApplicationInitParameter(String name);
028    
029        public Object getApplicationAttribute(String name);
030    
031        public void setApplicationAttribute(String name, Object value);
032    
033        public void removeApplicationAttribute(String name);
034    
035        public Object getPageAttribute(String name);
036    
037        public void setPageAttribute(String name, Object value);
038    
039        public void removePageAttribute(String name);
040    
041        public String getParameter(String name);
042    
043        public Map<?,?> getParameterMap();
044    
045        public void setParameterMap(Map<?,?> parameterMap);
046    
047        public Object getRequestAttribute(String name);
048    
049        public void setRequestAttribute(String name, Object value);
050    
051        public void removeRequestAttribute(String name);
052    
053        public Object getSessionAttribute(String name);
054    
055        public void setSessionAttribute(String name, Object value);
056    
057        public void removeSessionAttribute(String name);
058    
059        public Writer getWriter();
060    
061        public Locale getLocale();
062    
063        public void setLocale(Locale locale);
064    
065        public String getContextPath();
066        
067        public String getRealPath(String path);
068    
069        public Object getBackingObject();
070    }
