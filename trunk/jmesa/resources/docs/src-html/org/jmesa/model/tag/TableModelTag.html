454         * These are the converted items. What happens is the regular items are passed to the columns,
455         * but then the columns are resolved with the EL Expression Language. A new Map is swapped out
456         * for each row Bean (or Map) and then used in place of the regular items. So in effect this is
457         * a Collection of Maps that represents the original Collection of Beans (or Maps) after being
458         * processed by the columns.
459         * 
460         * @return The converted items.
461         */
462        Collection<Map<String, Object>> getPageItems() {
463            return pageItems;
464        }

466        /**
467         * Process the table, rows, and columns. Will pass the current bean in the pageScope for each
468         * bean in the Collection of Beans or Collection of Maps.
469         */
470        @Override
471        public void doTag() throws JspException, IOException {
472            JspFragment body = getJspBody();
473            if (body == null) {
474                throw new IllegalStateException("You need to wrap the table in the facade tag.");
475            }
476    
477            this.tableFacade = createTableFacade();
478            tableFacade.setPreferences(getTableFacadePreferences(getPreferences()));
479            tableFacade.setEditable(isEditable());
480            tableFacade.setItems(getItems());
481            tableFacade.setMaxRows(getMaxRows());
482            tableFacade.setStateAttr(getStateAttr());
483            tableFacade.setState(getTableFacadeState(getState()));
484            tableFacade.setLimit(getLimit());
485    
486            tableFacade.setMaxRowsIncrements(getTableFacadeMaxRowIncrements(getMaxRowsIncrements()));
487            tableFacade.setExportTypes(getTableFacadeExportTypes(getExportTypes()));
488    
489            tableFacade.autoFilterAndSort(isAutoFilterAndSort());
490            tableFacade.setColumnSort(getTableFacadeColumnSort(getColumnSort()));
491            tableFacade.setRowFilter(getTableFacadeRowFilter(getRowFilter()));
492            tableFacade.addFilterMatcherMap(getTableFacadeFilterMatcherMap(getFilterMatcherMap()));
493    
494            Collection<?> pi = tableFacade.getCoreContext().getPageItems();
495    
496            if (pi.size() == 0) {
497                body.invoke(null);
498                getPageItems().clear();
499            } else {
500                int count = HtmlUtils.startingRowcount(tableFacade.getCoreContext());
501                for (Iterator<?> iterator = pi.iterator(); iterator.hasNext();) {
502                    Object item = iterator.next();
503                    getJspContext().setAttribute(getVar(), item);
504                    getJspContext().setAttribute("rowcount", ++count);
505                    body.invoke(null);
506                }
507            }
508    
509            getJspContext().removeAttribute(getVar()); // clean up the page scoped bean
510            getJspContext().removeAttribute("rowcount");
511    
512            tableFacade.setTable(getTable());
513            tableFacade.setToolbar(getTableFacadeToolbar(getToolbar()));
514            tableFacade.setView(getTableFacadeView(getView()));
515            tableFacade.getCoreContext().setPageItems(getPageItems());
516    
517            Worksheet worksheet = tableFacade.getWorksheet();
518            if (ViewUtils.isEditable(worksheet)) {
519                    if (worksheet.isAddingRow()) {
520                            tableFacade.addWorksheetRow(getAddedRowObject());
521                    }
522                    
523                    if (worksheet.isRemovingRow()) {
524                            tableFacade.removeWorksheetRow();
525                    }
526            }
527    
528            View v = tableFacade.getView();
529            String html = v.render().toString();
530            getJspContext().getOut().print(html);
531        }
532    
533        protected WebContext getWebContext() {
534            return new JspPageWebContext((PageContext) getJspContext());
535        }
536    
537        protected TableFacade createTableFacade() {
538    
539            TableFacade facade = TableFacadeFactory.createTableFacade(getId(), getWebContext());
540            facade.setMessages(getTableFacadeMessages(getMessages()));
541            return facade;
542        }
