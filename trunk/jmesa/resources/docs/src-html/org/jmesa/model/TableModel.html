051        private Collection<?> items;
052        private PageItems pageItems;

054        private Preferences preferences;
055        private Messages messages;
056        private ExportType[] exportTypes;
057        private State state;

061        private Map<MatcherKey, FilterMatcher> filterMatchers;

064        private RowFilter rowFilter;
065        private int maxRows;
066        private int[] maxRowsIncrements;

070        private boolean editable;

072        private Object addedRowObject;

076        // only used to subclass the model
077        protected TableModel() {}

079        public TableModel(String id, HttpServletRequest request) {
080            this.id = id;
081            this.request = request;
082            this.tableFacade = new TableFacade(id, request);
083        }
084    
085        public TableModel(String id, HttpServletRequest request, HttpServletResponse response) {
086            this.id = id;
087            this.request = request;
088            this.tableFacade = new TableFacade(id, request, response);
089        }
090    
091        public TableModel(String id, WebContext webContext) {
092            this.tableFacade = new TableFacade(id, null);
093            tableFacade.setWebContext(webContext);
094            setHttpServletRequest(tableFacade);
095        }
096    
097        public TableModel(String id, WebContext webContext, HttpServletResponse response) {
098            this.tableFacade = new TableFacade(id, null, response);
099            tableFacade.setWebContext(webContext);
100            setHttpServletRequest(tableFacade);
101        }
102    
103        protected void setTableFacade(TableFacade tableFacade) {
104            this.tableFacade = tableFacade;
105            this.id = tableFacade.getId();
106            setHttpServletRequest(tableFacade);
107        }
108    
109        private void setHttpServletRequest(TableFacade tableFacade) {
110            Object backingObject = tableFacade.getWebContext().getBackingObject();
111            if (backingObject instanceof HttpServletRequest) {
112                request = (HttpServletRequest) backingObject;
113            }
114        }

168        public void addFilterMatcher(MatcherKey key, FilterMatcher matcher) {

177        }

198

216            this.worksheetSaver = worksheetSaver;

219        public boolean isExporting() {
220            return getExportType() != null;
221        }
222    
223        public ExportType getExportType() {
224            LimitActionFactory actionFactory = new LimitActionFactory(id, request.getParameterMap());
225            return actionFactory.getExportType();
226        }
227    
228        /**
229         * Generate the view.
230         *
231         * @return An html generated table will return the String markup. An export will be written out
232         *         to the response and this method will return null.
233         */
234        public String render() {
235            tableFacade.setEditable(editable);
236    
237            if (worksheetSaver != null) {
238                TableModelUtils.saveWorksheet(tableFacade, worksheetSaver);
239            }
240    
241            if (preferences != null) {
242                tableFacade.setPreferences(preferences);
243            }
244    
245            if (messages != null) {
246                tableFacade.setMessages(messages);
247            }
248    
249            if (exportTypes != null) {
250                tableFacade.setExportTypes(exportTypes);
251            }
252    
253            if (stateAttr != null) {
254                tableFacade.setStateAttr(stateAttr);
255            }
256    
257            if (state != null) {
258                tableFacade.setState(state);
259            }
260    
261            if (filterMatchers != null) {
262                for (Entry<MatcherKey, FilterMatcher> entry : filterMatchers.entrySet()) {
263                    tableFacade.addFilterMatcher(entry.getKey(), entry.getValue());
264                }
265            }
266    
267            if (filterMatcherMap != null) {
268                tableFacade.addFilterMatcherMap(filterMatcherMap);
269            }
270    
271            if (columnSort != null) {
272                tableFacade.setColumnSort(columnSort);
273            }
274    
275            if (rowFilter != null) {
276                tableFacade.setRowFilter(rowFilter);
277            }
278    
279            if (maxRows != 0) {
280                tableFacade.setMaxRows(maxRows);
281            }
282    
283            if (maxRowsIncrements != null) {
284                tableFacade.setMaxRowsIncrements(maxRowsIncrements);
285            }
286    
287            if (limit != null) {
288                tableFacade.setLimit(limit);
289            }
290    
291            if (pageItems != null) {
292                items = getItems(tableFacade, pageItems);
293            } else if (allItems != null) {
294                items = allItems.getItems();
295            }
296            
297            tableFacade.setItems(items);
298            tableFacade.autoFilterAndSort(autoFilterAndSort);
299    
300            if (table != null) {
301                tableFacade.setTable(table);
302            }
303            
304            if (toolbar != null) {
305                tableFacade.setToolbar(toolbar);
306            }
307    
308            if (view != null) {
309                tableFacade.setView(view);
310            }
311    
312            Worksheet worksheet = tableFacade.getWorksheet();
313            if (editable && worksheet.isAddingRow()) {
314                    if (addedRowObject != null) {
315                            tableFacade.addWorksheetRow(addedRowObject);
316                    } else {
317                            tableFacade.addWorksheetRow();
318                    }
319            }
320    
321            if (editable && worksheet.isRemovingRow()) {
322                tableFacade.removeWorksheetRow();
323            }
324    
325            return tableFacade.render();
326        }
