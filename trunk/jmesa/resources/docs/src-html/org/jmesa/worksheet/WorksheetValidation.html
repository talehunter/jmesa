016    package org.jmesa.worksheet;
017    
018    import org.apache.commons.lang.StringUtils;
019    import org.jmesa.view.AbstractContextSupport;
020    
021    /**
022     * @since 2.4.7
023     * @author Siddhant Agrawal
024     */
025    public class WorksheetValidation extends AbstractContextSupport {
026        public static final String TRUE = "true";
027    
028        private final String validationType;
029        private final String value;
030        private String errorMessage;
031        private String errorMessageKey;
032    
033        private boolean custom;
034    
035        /**
036         * WorksheetValidationType.REQUIRED, WorksheetValidation.TRUE
037         * WorksheetValidationType.EMAIL, WorksheetValidation.TRUE
038         * WorksheetValidationType.ACCEPT, "'jpg|png'"
039         * WorksheetValidationType.MAX_LENGTH, "10"
040         * WorksheetValidationType.MIN_LENGTH, "5"
041         * WorksheetValidationType.RANGE_LENGTH, "[5, 10]"
042         * WorksheetValidationType.RANGE, "[15, 20]"
043         * WorksheetValidationType.MAX_VALUE, "20"
044         * WorksheetValidationType.MIN_VALUE, "15"
045         *
046         * @since 2.4.7
047         */
048        public WorksheetValidation(String validationType, String value) {
049    
050            if (StringUtils.isEmpty(value)) {
051                throw new IllegalArgumentException("Value is required for worksheet validation: " + validationType);
052            }
053    
054            this.validationType = validationType;
055            this.value = value;
056        }
057    
058        public boolean isCustom() {
059            return custom;
060        }
061    
062        public void setCustom(boolean custom) {
063            this.custom = custom;
064        }
065    
066        public String getValidationType() {
067            return validationType;
068        }
069    
070        public void setErrorMessage(String errorMessage) {
071            this.errorMessage = errorMessage;
072        }
073    
074        public void setErrorMessageKey(String errorMessageKey) {
075            this.errorMessageKey = errorMessageKey;
076        }
077        
078        public WorksheetValidation errorMessage(String errorMessage) {
079            setErrorMessage(errorMessage);
080            return this;
081        }
082    
083        public WorksheetValidation errorMessageKey(String errorMessageKey) {
084            setErrorMessageKey(errorMessageKey);
085            return this;
086        }
087        
088        public String getRule() {
089            return this.validationType + ": " + this.value;
090        }
091        
092        public String getMessage() {
093    
094            String msg = errorMessage;
095            if (errorMessageKey != null) {
096                msg = getCoreContext().getMessage(errorMessageKey);
097            }
098    
099            if (msg == null) {
100                return "";
101            }
102    
103            return this.validationType + ": '" + msg + "'";
104        }
105        
106        public String getCustomWorksheetValidation() {
107            if (!custom) {
108                    return "";
109            }
110            
111            return "jQuery.validator.addMethod('" + validationType + "', " + value + ");\n";
112        }
113    }
