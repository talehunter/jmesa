016    package org.jmesa.view;
017    
018    import java.util.List;
019    
020    import org.apache.commons.lang.StringUtils;
021    import org.jmesa.limit.ExportType;
022    import org.jmesa.view.html.component.HtmlColumn;
023    import org.jmesa.worksheet.Worksheet;
024    
025    /**
026     * @since 2.0
027     * @author Jeff Johnston
028     */
029    public class ViewUtils {
030        private ViewUtils() {}
031    
032        /**
033         * Convert camelCase text to a readable word. example: camelCaseToWord --> Camel Case To Word
034         */
035        public static String camelCaseToWord(String camelCaseText) {
036            if (StringUtils.isEmpty(camelCaseText)) {
037                return camelCaseText;
038            }
039    
040            if (camelCaseText.equals(camelCaseText.toUpperCase())) {
041                return camelCaseText;
042            }
043    
044            char[] ch = camelCaseText.toCharArray();
045            String first = "" + ch[0];
046            String build = first.toUpperCase();
047    
048            for (int i = 1; i < ch.length; i++) {
049                String test = "" + ch[i];
050    
051                if (test.equals(test.toUpperCase())) {
052                    build += " ";
053                }
054    
055                build += test;
056            }
057    
058            return build;
059        }
060    
061        /**
062         * Find out if the column is sitting on an even row.
063         */
064        public static boolean isRowEven(int rowcount) {
065            if (rowcount != 0 && (rowcount % 2) == 0) {
066                return true;
067            }
068    
069            return false;
070        }
071    
072        /**
073         * Find out if the column is sitting on an odd row.
074         */
075        public static boolean isRowOdd(int rowcount) {
076            if (rowcount != 0 && (rowcount % 2) == 0) {
077                return false;
078            }
079    
080            return true;
081        }
082    
083        /**
084         * @return Is true if any columns are filterable.
085         */
086        public static boolean isFilterable(List<HtmlColumn> columns) {
087            for (HtmlColumn column : columns) {
088                if (column.isFilterable()) {
089                    return true;
090                }
091            }
092    
093            return false;
094        }
095        
096        /**
097         * @param exportTypes The array of export types to check.
098         * @return Is true if there is an export to do.
099         */
100        public static boolean isExportable(ExportType... exportTypes) {
101            return  exportTypes != null && exportTypes.length > 0;
102        }
103        
104        /**
105         * @return Is true if there is a worksheet.
106         */
107        public static boolean isEditable(Worksheet worksheet) {
108            if (worksheet != null) {
109                return true;
110            }
111    
112            return false;
113        }    
114    }
