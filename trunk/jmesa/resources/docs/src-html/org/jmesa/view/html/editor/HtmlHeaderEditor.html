016    package org.jmesa.view.html.editor;
017    
018    import org.jmesa.limit.Limit;
019    import org.jmesa.limit.Order;
020    import org.jmesa.limit.Sort;
021    import org.jmesa.view.editor.AbstractHeaderEditor;
022    import org.jmesa.view.html.HtmlBuilder;
023    import org.jmesa.view.html.HtmlConstants;
024    import org.jmesa.view.html.HtmlUtils;
025    import org.jmesa.view.html.component.HtmlColumn;
026    import org.jmesa.view.html.toolbar.ToolbarItemType;
027    
028    /**
029     * The default editor for the column header. Will handle all the sorting.
030     * 
031     * @since 2.2
032     * @author Jeff Johnston
033     */
034    public class HtmlHeaderEditor extends AbstractHeaderEditor {
035        @Override
036        public HtmlColumn getColumn() {
037            return (HtmlColumn) super.getColumn();
038        }
039    
040        public Object getValue() {
041            HtmlBuilder html = new HtmlBuilder();
042            
043            html.div();
044    
045            Limit limit = getCoreContext().getLimit();
046            HtmlColumn column = getColumn();
047    
048            if (column.isSortable()) {
049                Sort sort = limit.getSortSet().getSort(column.getProperty());
050    
051                html.onmouseover("this.style.cursor='pointer'");
052                html.onmouseout("this.style.cursor='default'");
053    
054                if (sort != null) {
055                    Order nextOrder = nextSortOrder(sort.getOrder(), column);
056                    html.append(onclick(nextOrder, column, limit));
057                } else {
058                    Order[] sortOrder = column.getSortOrder();
059                    if (sortOrder[0] == Order.NONE) {
060                        Order nextOrder = nextSortOrder(sortOrder[0], column);
061                        html.append(onclick(nextOrder, column, limit));
062                    } else {
063                        Order nextOrder = sortOrder[0];
064                        html.append(onclick(nextOrder, column, limit));
065                    }
066                }
067            }
068            
069            html.close();
070            html.append(getTitle(column));
071    
072            if (column.isSortable()) {
073                String imagesPath = HtmlUtils.imagesPath(getWebContext(), getCoreContext());
074                Sort sort = limit.getSortSet().getSort(column.getProperty());
075                if (sort != null) {
076                    if (sort.getOrder() == Order.ASC) {
077                        html.nbsp();
078                        html.img();
079                        html.src(imagesPath + getCoreContext().getPreference(HtmlConstants.SORT_ASC_IMAGE));
080                        html.style("border:0");
081                        html.alt("Arrow");
082                        html.end();
083                    } else if (sort.getOrder() == Order.DESC) {
084                        html.nbsp();
085                        html.img();
086                        html.src(imagesPath + getCoreContext().getPreference(HtmlConstants.SORT_DESC_IMAGE));
087                        html.style("border:0");
088                        html.alt("Arrow");
089                        html.end();
090                    }
091                } else {
092                    String sortDefaultImage = getCoreContext().getPreference(HtmlConstants.SORT_DEFAULT_IMAGE);
093                    if (sortDefaultImage != null && sortDefaultImage.length() > 0) {
094                        html.nbsp();
095                        html.img();
096                        html.src(imagesPath + sortDefaultImage);
097                        html.style("border:0");
098                        html.alt("Arrow");
099                        html.end();
100                    }
101                }
102            }
103            
104            html.divEnd();
105    
106            return html.toString();
107        }
108    
109        protected String getTitle(HtmlColumn column) {
110            return column.getTitle();
111        }
112    
113        /**
114         * @param currentOrder The current sort Order.
115         * @param column The current column.
116         * @param limit The current limit.
117         * @return The JavaScript to get the next Order when invoking the onlick command.
118         */
119        protected String onclick(Order currentOrder, HtmlColumn column, Limit limit) {
120            HtmlBuilder html = new HtmlBuilder();
121    
122            int position = column.getRow().getColumns().indexOf(column);
123    
124            if (currentOrder == Order.NONE) {
125                String onInvokeAction = getCoreContext().getPreference(HtmlConstants.ON_INVOKE_ACTION);
126                html.onclick(
127                    "jQuery.jmesa.removeSortFromLimit('" + limit.getId() + "','" + 
128                    column.getProperty() + "');" + onInvokeAction + "('" + limit.getId() + "')"
129                );
130            } else {
131                html.onclick(
132                    "jQuery.jmesa.addSortToLimit('" + limit.getId() + "','" + 
133                    position + "','" + column.getProperty() + "','" + 
134                    currentOrder.toParam() + "');" + getOnInvokeActionJavaScript(limit)
135                );
136            }
137    
138            return html.toString();
139        }
140    
141        /**
142         * @param currentOrder The current sort Order.
143         * @param column The current column.
144         * @return The next sort order in the array.
145         */
146        protected Order nextSortOrder(Order currentOrder, HtmlColumn column) {
147            Order[] sortOrder = column.getSortOrder();
148    
149            for (int i = 0; i < sortOrder.length; i++) {
150                Order order = sortOrder[i];
151                if (order == currentOrder) {
152                    if (i + 1 == sortOrder.length) {
153                        return sortOrder[0];
154                    }
155    
156                    return sortOrder[i + 1];
157                }
158            }
159    
160            return Order.NONE;
161        }
162        
163        protected String getOnInvokeActionJavaScript(Limit limit) {
164            String onInvokeAction = getCoreContext().getPreference(HtmlConstants.ON_INVOKE_ACTION);
165            return onInvokeAction + "('" + limit.getId() + "', '" + ToolbarItemType.SORT_ITEM.toCode() + "')";
166        }
167    }
