016    package org.jmesa.view.html;
017    
018    import org.jmesa.core.CoreContext;
019    import org.jmesa.view.AbstractComponentFactory;
020    import org.jmesa.view.editor.CellEditor;
021    import org.jmesa.view.html.component.HtmlColumn;
022    import org.jmesa.view.html.component.HtmlRow;
023    import org.jmesa.view.html.component.HtmlTable;
024    import org.jmesa.view.html.editor.HtmlCellEditor;
025    import org.jmesa.view.html.editor.HtmlFilterEditor;
026    import org.jmesa.view.html.editor.HtmlHeaderEditor;
027    import org.jmesa.view.html.renderer.HtmlCellRendererImpl;
028    import org.jmesa.view.html.renderer.HtmlFilterRendererImpl;
029    import org.jmesa.view.html.renderer.HtmlHeaderRendererImpl;
030    import org.jmesa.view.html.renderer.HtmlRowRendererImpl;
031    import org.jmesa.view.html.renderer.HtmlTableRendererImpl;
032    import org.jmesa.web.WebContext;
033    
034    /**
035     * @since 2.0
036     * @author Jeff Johnston
037     *
038     * @deprecated Should build components directly now instead of using factory.
039     */
040    @Deprecated
041    public class HtmlComponentFactory extends AbstractComponentFactory {
042        public HtmlComponentFactory(WebContext webContext, CoreContext coreContext) {}
043    
044        /**
045         * @deprecated Should build components directly now instead of using factory.
046         */
047        @Deprecated
048        @Override
049        public HtmlTable createTable() {
050            HtmlTable table = new HtmlTable();
051    
052            HtmlTableRendererImpl tableRenderer = new HtmlTableRendererImpl(table);
053            table.setTableRenderer(tableRenderer);
054    
055            return table;
056        }
057    
058        /**
059         * @deprecated Should build components directly now instead of using factory.
060         */
061        @Deprecated
062        @Override
063        public HtmlRow createRow() {
064            HtmlRow row = new HtmlRow();
065    
066            HtmlRowRendererImpl rowRenderer = new HtmlRowRendererImpl(row);
067            row.setRowRenderer(rowRenderer);
068    
069            return row;
070        }
071    
072        /**
073         * @deprecated Should build components directly now instead of using factory.
074         */
075        @Deprecated
076        @Override
077        public CellEditor createBasicCellEditor() {
078            HtmlCellEditor editor = new HtmlCellEditor();
079            return editor;
080        }
081    
082        /**
083         * Create a column using the BasicCellEditor.
084         * 
085         * @param property The column property.
086         * @return The HtmlColumn instance.
087         * @deprecated Should build components directly now instead of using factory.
088         */
089        @Deprecated
090        @Override
091        public HtmlColumn createColumn(String property) {
092            return createColumn(property, createBasicCellEditor());
093        }
094    
095        /**
096         * Create a column that does not require cell editor.
097         * 
098         * @return The HtmlColumn instance.
099         * @deprecated Should build components directly now instead of using factory.
100         */
101        @Deprecated
102        @Override
103        public HtmlColumn createColumn(CellEditor editor) {
104            return createColumn(null, editor);
105        }
106    
107        /**
108         * @deprecated Should build components directly now instead of using factory.
109         */
110        @Deprecated
111        public HtmlColumn createColumn(String property, CellEditor editor) {
112            HtmlColumn column = new HtmlColumn(property);
113            
114            // cell
115    
116            HtmlCellRendererImpl cellRenderer = new HtmlCellRendererImpl(column, editor);
117            column.setCellRenderer(cellRenderer);
118    
119            // filter
120            
121            HtmlFilterRendererImpl filterRenderer = new HtmlFilterRendererImpl(column);
122            column.setFilterRenderer(filterRenderer);
123    
124            HtmlFilterEditor filterEditor = new HtmlFilterEditor();
125            filterEditor.setColumn(column);
126            filterRenderer.setFilterEditor(filterEditor);
127            
128            // header
129    
130            HtmlHeaderRendererImpl headerRenderer = new HtmlHeaderRendererImpl(column);
131            column.setHeaderRenderer(headerRenderer);
132    
133            HtmlHeaderEditor headerEditor = new HtmlHeaderEditor();
134            headerEditor.setColumn(column);
135            headerRenderer.setHeaderEditor(headerEditor);
136    
137            return column;
138        }
139    }
