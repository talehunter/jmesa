016    package org.jmesa.view.pdf;

018    import java.util.Iterator;
019    import java.util.List;
020    import org.apache.commons.lang.StringEscapeUtils;
021    import org.apache.commons.lang.StringUtils;
022    import org.jmesa.util.SupportUtils;
023    import org.jmesa.view.AbstractExportView;

025    import org.jmesa.view.editor.CellEditor;
026    import org.jmesa.view.html.HtmlBuilder;
027    import org.jmesa.view.html.HtmlSnippets;
028    import org.jmesa.view.html.HtmlSnippetsImpl;
029    import org.jmesa.view.html.component.HtmlColumn;
030    import org.jmesa.view.html.component.HtmlRow;
031    import org.jmesa.view.html.component.HtmlTable;
032    import org.jmesa.view.html.renderer.HtmlCellRenderer;
033    import org.slf4j.Logger;
034    import org.slf4j.LoggerFactory;
035    
036    /**
037     * Use the Flying Saucer API to generate Pdf documents.
038     *
039     * @since 2.2
040     * @author Paul Horn
041     */
042    public class PdfView extends AbstractExportView {
043        private static Logger logger = LoggerFactory.getLogger(PdfView.class);
044        
045        private String cssLocation;
046    
047        /**
048         * @return The stylesheet to use for this pdf.
049         */
050        public String getCssLocation() {
051            if (StringUtils.isEmpty(cssLocation)) {
052                cssLocation = getCoreContext().getPreference("pdf.cssLocation");
053            }
054            return cssLocation;
055        }
056    
057        /**
058         * <p>
059         * The css to use for this pdf file. Will be relative to the servlet context.
060         * </p>
061         *
062         * <p>
063         * example: /css/jmesa-pdf-landscape.css
064         * <p>
065         *
066         * @param cssLocation The path and name of the jmesa css file.
067         */
068        public void setCssLocation(String cssLocation) {
069            this.cssLocation = cssLocation;
070        }
071    
072        public Object render() {
073            HtmlBuilder html = new HtmlBuilder();
074    
075            String contextPath = getWebContext().getContextPath();
076    
077            html.append(getCoreContext().getPreference("pdf.doctype"));
078    
079            html.html().close();
080    
081            html.head().close();
082    
083            html.link().rel("stylesheet").type("text/css").href(contextPath + getCssLocation()).media("print").end();
084    
085            html.headEnd();
086    
087            html.body().close();
088    
089            HtmlSnippets snippets = new HtmlSnippetsImpl((HtmlTable)getTable(), null, getCoreContext());
090            SupportUtils.setWebContext(snippets, getWebContext());
091    
092            html.append(snippets.themeStart());
093    
094            html.append(snippets.tableStart());
095    
096            html.append(snippets.theadStart());
097    
098            html.append(snippets.header());
099    
100            html.append(snippets.theadEnd());
101    
102            html.append(snippets.tbodyStart());
103    
104            decorateCellEditors();
105    
106            html.append(snippets.body());
107    
108            html.append(snippets.tbodyEnd());
109    
110            html.append(snippets.footer());
111    
112            html.append(snippets.statusBar());
113    
114            html.append(snippets.tableEnd());
115    
116            html.append(snippets.themeEnd());
117    
118            html.bodyEnd();
119    
120            html.htmlEnd();
121    
122            return html.toString();
123        }
124    
125        /**
126         * Decorate the cellEditors to escape xml values.
127         */
128        protected void decorateCellEditors() {
129            HtmlTable table = (HtmlTable) getTable();
130            HtmlRow row = table.getRow();
131            List<Column> columns = row.getColumns();
132            for (Iterator<Column> iter = columns.iterator(); iter.hasNext();) {
133                HtmlColumn column = (HtmlColumn) iter.next();
134                HtmlCellRenderer cellRenderer = column.getCellRenderer();
135                CellEditor cellEditor = cellRenderer.getCellEditor();
136    
137                // decorate the cell editors
138                EscapeXmlCellEditor escapeXmlCellEditor = new EscapeXmlCellEditor();
139                escapeXmlCellEditor.setCellEditor(cellEditor);
140                cellRenderer.setCellEditor(escapeXmlCellEditor);
141            }
142        }
143    
144        /**
145         * Decorate the cell editor with one that can escape values.
146         */
147        private static class EscapeXmlCellEditor implements CellEditor {
148    
149            private CellEditor cellEditor;
150    
151            public void setCellEditor(CellEditor cellEditor) {
152                this.cellEditor = cellEditor;
153            }
154    
155            public Object getValue(Object item, String property, int rowcount) {
156                Object value = cellEditor.getValue(item, property, rowcount);
157                if (value != null) {
158                    return StringEscapeUtils.escapeXml(value.toString());
159                }
160    
161                return null;
162            }
163        }
164    }
