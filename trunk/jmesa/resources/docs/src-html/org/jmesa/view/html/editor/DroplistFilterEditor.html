018    import static org.apache.commons.lang.StringEscapeUtils.escapeHtml;
019    import static org.apache.commons.lang.StringEscapeUtils.escapeJavaScript;
020    
021    import java.util.ArrayList;
022    import java.util.Collection;
023    import java.util.Collections;
024    import java.util.HashSet;
025    import java.util.Iterator;
026    import java.util.List;
027    import java.util.Set;
028    
029    import org.apache.commons.lang.builder.ToStringBuilder;
030    import org.jmesa.limit.Filter;
031    import org.jmesa.limit.Limit;

037    /**
038     * Create a droplist for the filter.
039     * 
040     * @since 2.2
041     * @author Jeff Johnston
042     */
043    public class DroplistFilterEditor extends AbstractFilterEditor {
044        private Set<Option> options;
045        
046        public Object getValue() {
047            HtmlBuilder html = new HtmlBuilder();
048    
049            Limit limit = getCoreContext().getLimit();
050            HtmlColumn column = (HtmlColumn) getColumn();
051            String property = column.getProperty();
052            Filter filter = limit.getFilterSet().getFilter(property);
053    
054            String filterValue = "";
055            if (filter != null) {
056                filterValue = filter.getValue();
057            }
058    
059            String filterLabel = "";
060    
061            StringBuilder array = new StringBuilder();
062            array.append("{");
063    
064            Collection<Option> opts = getOptions();
065            for (Iterator<Option> it = opts.iterator(); it.hasNext();) {
066                Option option = it.next();
067    
068                String value = option.getValue();
069                String label = option.getLabel();
070    
071                if (value.equalsIgnoreCase(filterValue)){
072                    filterLabel = label;
073                }
074    
075                value = escapeJavaScript(escapeHtml(value));
076                label = escapeJavaScript(escapeHtml(label));
077    
078                array.append("'").append(value).append("':'").append(label).append("'");
079                if (it.hasNext()) {
080                    array.append(",");
081                }
082            }
083    
084            array.append("}");
085    
086            html.div().styleClass("dynFilter");
087            html.onclick("jQuery.jmesa.createDroplistDynFilter(this,'" + limit.getId() + "','" + column.getProperty() + "'," + array + ")");
088            html.close();
089            html.append(escapeHtml(filterLabel));
090            html.divEnd();
091    
092            return html.toString();
093        }
094        
095        public void addOptions(Collection<?> beans, String valueProperty, String labelProperty) {
096            for (Object bean : beans) {
097                addOption(bean, valueProperty, labelProperty);
098            }
099        }
100    
101        public void addOption(Object bean, String valueProperty, String labelProperty) {
102            Object value = ItemUtils.getItemValue(bean, valueProperty);
103            Object label = ItemUtils.getItemValue(bean, labelProperty);
104            addOption(
105                value == null ? null : String.valueOf(value),
106                label == null ? null : String.valueOf(label)        
107            );
108        }
109        
110        public void addOption(String value, String label) {
111            if (value == null || value.length() == 0) {
112                return;
113            }
114            if (label == null || label.length() == 0) {
115                label = value;
116            }
117            this.addOption(new Option(value, label));
118        }
119        
120        protected void addOption(Option option) {
121            if (options == null) {
122                options = new HashSet<Option>();
123            }
124            options.add(option);
125        }
126        
127        /**
128         * @return The unique list of options for the droplist values.
129         */
130        protected Collection<Option> getOptions() {
131    
132            List<Option> opts;
133    
134            if (this.options == null) {
135                Set<String> values = new HashSet<String>();
136                String property = getColumn().getProperty();
137                for (Object item : getCoreContext().getAllItems()) {
138                    Object value = ItemUtils.getItemValue(item, property);
139                    if (value == null) {
140                        continue;
141                    }
142                    String valueStr = String.valueOf(value);
143                    if (valueStr.length() == 0) {
144                        continue;
145                    }
146                    values.add(valueStr);
147                }
148                opts = new ArrayList<Option>();
149                for (String value : values) {
150                    Option option = new Option(value, value);
151                    opts.add(option);
152                }
153            } else {
154                opts = new ArrayList<Option>(this.options);
155            }
156    
157            Collections.sort(opts, null);
158    
159            return opts;
160        }
161    
162        /**
163         * Represents an Html Select Option.
164         */
165        protected static class Option implements Comparable<Option> {
166            private final String value;
167            private final String label;
168    
169            public Option(String value, String label) {
170                this.value = value;
171                this.label = label;
172            }
173    
174            public String getValue() {
175                return value;
176            }
177    
178            public String getLabel() {
179                return label;
180            }
181    
182            public int compareTo(Option option) {
183                return this.getLabel().compareToIgnoreCase(option.getLabel());
184            }
185    
186            /**
187             * Equality is based on the label. Or, in other words no two Option
188             * Objects can have the same property.
189             */
190            @Override
191            public boolean equals(Object o) {
192                if (o == this) {
193                    return true;
194                }
195    
196                if (!(o instanceof Option)) {
197                    return false;
198                }
199    
200                Option that = (Option) o;
201    
202                return that.getLabel().equals(this.getLabel());
203            }
204    
205            @Override
206            public int hashCode() {
207                int result = 17;
208                int property = this.getLabel() == null ? 0 : this.getLabel().hashCode();
209                result = result * 37 + property;
210                return result;
211            }
212    
213            @Override
214            public String toString() {
215                ToStringBuilder builder = new ToStringBuilder(this);
216                builder.append("value", getValue());
217                builder.append("label", getLabel());
218                return builder.toString();
219            }
220        }
221    }
