016    package org.jmesa.view.html.editor;
017    
018    import static org.apache.commons.lang.StringEscapeUtils.escapeHtml;
019    import org.jmesa.limit.Filter;
020    import org.jmesa.limit.Limit;
021    import org.jmesa.view.editor.AbstractFilterEditor;
022    import org.jmesa.view.html.HtmlBuilder;
023    import org.jmesa.view.html.component.HtmlColumn;
024    
025    /**
026     * The default editor for the column filter.
027     * 
028     * @since 2.2
029     * @author Jeff Johnston
030     */
031    public class HtmlFilterEditor extends AbstractFilterEditor {
032        @Override
033        public HtmlColumn getColumn() {
034            return (HtmlColumn) super.getColumn();
035        }
036    
037        public Object getValue() {
038            HtmlBuilder html = new HtmlBuilder();
039    
040            Limit limit = getCoreContext().getLimit();
041            HtmlColumn column = getColumn();
042            String property = column.getProperty();
043            Filter filter = limit.getFilterSet().getFilter(property);
044    
045            String filterValue = "";
046            if (filter != null) {
047                filterValue = filter.getValue();
048            }
049    
050            html.div().styleClass("dynFilter");
051            html.onclick("jQuery.jmesa.createDynFilter(this, '" + limit.getId() + "','" + column.getProperty() + "')");
052            html.close();
053            html.append(escapeHtml(filterValue));
054            html.divEnd();
055    
056            return html.toString();
057        }
058    }
