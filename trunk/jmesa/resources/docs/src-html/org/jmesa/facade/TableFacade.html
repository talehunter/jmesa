089    /**
090     * <p>
091     * This is a facade for working with tables that also has a little bit of builder in its veins. The
092     * basic idea is you instantiate a TableFacade class and then interact with it in a natural way. The
093     * facade completely abstracts away all the factory classes and eliminates a lot of boilerplate
094     * code. The builder notion comes in because as you work with it it will internally build up
095     * everything you need and keep track of it for you.
096     * </p>
097     *
098     * <p>
099     * Notice how there are no factories to deal with. However any API Object that you would have used
100     * before is available through the facade, including the WebContext, CoreContext, Limit, Table,
101     * Toolbar, and View. When you ask the facade for a given object it builds everything it needs up to
102     * that point. Internally it keeps track of everything you are doing so it also works like a
103     * builder.
104     * </p>
105     *
106     * <p>
107     * The TableFacade also has setters for all the facade objects including the WebContext,
108     * CoreContext, Limit, Table, Toolbar, and View. The reason is if you really need to customize
109     * something and want to set your own implementation you can. Your object just goes into the flow of
110     * the facade. For instance if you want a custom toolbar just set the Toolbar on the facade and when
111     * the render() method is called it will use your Toolbar.
112     * </p>
113     *
114     * <p>
115     * However, all this should feel very natural and you should not have to think about what you are
116     * doing. Just interact with the facade how you need to and it will take care of everything.
117     * </p>
118     *
119     * @since 2.1
120     * @author Jeff Johnston
121     */

251                return worksheet;
252            }
253    
254            this.worksheetState = getWorksheetState();
255            this.worksheet = worksheetState.retrieveWorksheet();
256    
257            if (worksheet == null || isClearingWorksheet(id, getWebContext()) || !isTableRefreshing(id, getWebContext())) {

259                persistWorksheet(worksheet);
260            }
261    
262            worksheet.setWebContext(getWebContext());
263            worksheet.setMessages(getMessages());
264    
265            return worksheet;

309        /**
310         * Remove a worksheet row.
311         */
312        public void removeWorksheetRow() {
313            String up = getLimit().getId() + "_" +  Worksheet.REMOVE_WORKSHEET_ROW;
314            String name = getTable().getRow().getUniqueProperty();
315            String value = getWebContext().getParameter(up);
316            UniqueProperty uniqueProperty = new UniqueProperty(name, value);
317    
318            Worksheet ws = getWorksheet();
319            WorksheetRow wsRow = ws.getRow(uniqueProperty);
320    
321            if (wsRow != null) {
322                if (wsRow.getRowStatus().equals(WorksheetRowStatus.ADD)) {
323                    // remove row if ADDED in worksheet
324                    ws.removeRow(wsRow);
325                } else if (wsRow.getRowStatus().equals(WorksheetRowStatus.REMOVE)) {
326                    // undo - remove
327                    if (wsRow.getColumns().size() == 0) {
328                        ws.removeRow(uniqueProperty);
329                    } else {
330                        wsRow.setRowStatus(WorksheetRowStatus.MODIFY);
331                    }
332                } else {
333                    wsRow.setRowStatus(WorksheetRowStatus.REMOVE);
334    
335                    boolean keepChangedValues = Boolean.parseBoolean(
336                            getCoreContext().getPreference(HtmlConstants.REMOVE_ROW_KEEP_CHANGED_VALUES));
337                    if (!keepChangedValues) {
338                        wsRow.getColumns().clear();
339                    }
340                }
341            } else {
342                // add a new REMOVED row in worksheet
343                wsRow = new WorksheetRow(uniqueProperty);
344                wsRow.setRowStatus(WorksheetRowStatus.REMOVE);
345                ws.addRow(wsRow);
346            }
347            // for GAE
348            persistWorksheet(ws);
349        }

351        /**
352         * <p>
353         * Get the Limit. If the Limit does not exist then one will be created. If you are manually
354         * sorting and filtering the table then as much of the Limit will be created as is possible. You
355         * still might need to set the totalRows on the facade, which will set it on the Limit.
356         * </p>
357         *
358         * <p>
359         * If using the State interface then be sure to set that on the facade before requesting the Limit.
360         * </p>
361         */
362        public Limit getLimit() {
363            if (limit != null) {
364                return limit;
365            }
366    
367            LimitFactory limitFactory = new LimitFactory(id, getWebContext());
368            limitFactory.setState(getState());
369            this.limit = limitFactory.createLimit();
370    
371            if (items == null) {
372                return limit;
373            }
374    
375            /*
376             * We will have a RowSelect if the State returned a Limit. However,
377             * we will still need to re-populate the Limit's RowSelect based
378             * on what the current items total rowcount is.
379             *
380             * If we do not have a RowSelect then we need to populate the Limit's
381             * RowSelect based on what the current items total rowcount is.
382             */
383            if (limit.hasRowSelect()) {
384                int p = limit.getRowSelect().getPage();
385                int mr = limit.getRowSelect().getMaxRows();
386                RowSelect rowSelect = new RowSelect(p, mr, items.size());
387                limit.setRowSelect(rowSelect);
388            } else if (limit.hasExport()) {
389                RowSelect rowSelect = new RowSelect(1, items.size(), items.size());
390                limit.setRowSelect(rowSelect);
391            } else {
392                RowSelect rowSelect = limitFactory.createRowSelect(getMaxRows(), items.size());
393                limit.setRowSelect(rowSelect);
394            }
395    
396            return limit;
397        }

544        /**
545         * <p>
546         * Add a FilterMatcherMap on the facade. Will add the various FilterMatchers to the facade using
547         * the FilterMatcherMap interface.
548         * </p>
549         *
550         * <p>
551         * Most useful for the tag library because they have to use the FilterMatcherMap to register
552         * filter matcher strategies, but could also be used as a way to bundle up filter matchers.
553         * </p>
554         */
555        public void addFilterMatcherMap(FilterMatcherMap filterMatcherMap) {
556            validateCoreContextIsNull(coreContext, "FilterMatcher");
557    
558            if (filterMatcherMap == null) {
559                return;
560            }
561    
562            SupportUtils.setWebContext(filterMatcherMap, getWebContext());
563    
564            Map<MatcherKey, FilterMatcher> matchers = filterMatcherMap.getFilterMatchers();
565            Set<MatcherKey> keys = matchers.keySet();
566            for (MatcherKey key : keys) {
567                FilterMatcher matcher = matchers.get(key);
568                addFilterMatcher(key, matcher);
569            }
570        }

643        /**
644         * Get the CoreContext. If the CoreContext does not exist then one will be created.
645         */
646        public CoreContext getCoreContext() {
647            if (coreContext != null) {
648                return coreContext;
649            }
650    
651            validateItemsIsNotNull(items);
652    
653            CoreContextFactory factory = new CoreContextFactory(autoFilterAndSort, getWebContext());
654            factory.setPreferences(getPreferences());
655            factory.setMessages(getMessages());
656            factory.setColumnSort(columnSort);
657            factory.setRowFilter(rowFilter);
658    
659            if (filterMatchers != null) {
660                Set<MatcherKey> keySet = filterMatchers.keySet();
661                for (MatcherKey key : keySet) {
662                    FilterMatcher matcher = filterMatchers.get(key);
663                    factory.addFilterMatcher(key, matcher);
664                }
665            }
666    
667            this.coreContext = factory.createCoreContext(items, getLimit(), getWorksheet());
668            return coreContext;
669        }

681        /**
682         * Get the Table. If the Table does not exist then one will be created.
683         */
684        public Table getTable() {
685            //TODO: the follow code should be removed when the 3.0 deprecated methods are removed.
686            if (table == null) {
687                validateColumnPropertiesIsNotNull(columnProperties);
688    
689                Limit l = getLimit();
690    
691                if (!l.isExported()) {
692                    validateRowSelectIsNotNull(l);
693    
694                    HtmlTableFactory tableFactory = new HtmlTableFactory();
695                    this.table = tableFactory.createTable(columnProperties);
696                } else {
697                    this.table = getExportTable(l.getExportType());
698                }
699            }
700    
701            TableFacadeUtils.initTable(this, table);
702            
703            return table;
704        }

708            //TODO: the follow code should be removed when the 3.0 deprecated methods are removed.

710            TableFactory tableFactory = null;
711    
712            if (exportType == ExportType.CSV) {
713                tableFactory = new CsvTableFactory();
714            } else if (exportType == ExportType.EXCEL) {
715                tableFactory = new ExportTableFactory();
716            } else if (exportType == ExportType.JEXCEL) {
717                tableFactory = new ExportTableFactory();
718            } else if (exportType == ExportType.PDF) {
719                tableFactory = new HtmlTableFactory();
720            } else if (exportType == ExportType.PDFP) {
721                tableFactory = new ExportTableFactory();
722            }
723    
724            if (tableFactory != null) {
725                return tableFactory.createTable(columnProperties);
726            }
727    
728            throw new IllegalStateException("Not able to handle the export of type: " + exportType);
729        }

786            if (view != null) {
787                return view;
788            }
789    
790            Limit l = getLimit();
791    
792            if (!l.isExported()) {
793                setView(new HtmlView());
794            } else {
795                this.view = getExportView(l.getExportType());
796            }
797    
798            return view;

801        protected View getExportView(ExportType exportType) {
802    
803            View exportView = null;
804    
805            if (exportType == ExportType.CSV) {
806                exportView = new CsvView(",");
807            } else if (exportType == ExportType.EXCEL) {
808                exportView = new ExcelView();
809            } else if (exportType == ExportType.JEXCEL) {
810                exportView = new JExcelView();
811            } else if (exportType == ExportType.PDF) {
812                exportView = new PdfView();
813            } else if (exportType == ExportType.PDFP) {
814                exportView= new PdfPView();
815            }
816    
817            if (exportView != null) {
818                SupportUtils.setCoreContext(exportView, getCoreContext());
819                SupportUtils.setWebContext(exportView, getWebContext());
820                SupportUtils.setTable(exportView, getTable());
821                return exportView;
822            }
823    
824            throw new IllegalStateException("Not able to handle the export of type: " + exportType);
825        }

838        /**
839         * Generate the view.
840         *
841         * @return An html generated table will return the String markup. An export will be written out
842         *         to the response and this method will return null.
843         */
844        public String render() {
845            Limit l = getLimit();
846            View v = getView();
847    
848            if (!l.isExported()) {
849                return v.render().toString();
850            }
851    
852            ExportType exportType = l.getExportType();
853            renderExport(exportType, v);
854    
855            return null;
856        }

858        protected void renderExport(ExportType exportType, View view) {
859            if (response == null) {
860                    throw new IllegalStateException(
861                        "The HttpServletResponse is null. You need to call the " +
862                        "TableFacadeImpl constructor (or factory) with the response object.");
863            }
864    
865            try {
866                CoreContext cc = getCoreContext();
867    
868                if (exportType == ExportType.CSV) {
869                    new CsvViewExporter(view, cc, response).export();
870                } else if (exportType == ExportType.EXCEL) {
871                    new ExcelViewExporter(view, cc, response).export();
872                } else if (exportType == ExportType.JEXCEL) {
873                    new JExcelViewExporter(view, cc, response).export();
874                } else if (exportType == ExportType.PDF) {
875                    new PdfViewExporter(view, cc, request, response).export();
876                } else if (exportType == ExportType.PDFP) {
877                    new PdfPViewExporter(view, cc, request, response).export();
878                }
879            } catch (Exception e) {
880                logger.error("Not able to perform the " + exportType + " export.", e);
881            }
882        }
